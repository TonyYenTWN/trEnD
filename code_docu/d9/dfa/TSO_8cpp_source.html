215void power_market::TSO_Market_Scheduled_Results_Get(int tick, market_inform &Market){
216 alglib::real_1d_array sol;
217 alglib::minlpreport rep;
218 alglib::minlpresults(Market.Problem, sol, rep);
219 Eigen::VectorXd sol_vec = Eigen::Map <Eigen::VectorXd> (sol.getcontent(), sol.length());
220
221 for(int node_iter = 0; node_iter < Market.network.num_vertice; ++ node_iter){
222 // Store power source / sink
223 int row_start = 2 * Market.network.num_vertice + node_iter * (Market.price_intervals + 2);
224 Market.confirmed.supply(tick, node_iter) = (sol_vec.segment(row_start, Market.price_intervals + 2).array().max(0)).sum();
225 Market.confirmed.demand(tick, node_iter) = -(sol_vec.segment(row_start, Market.price_intervals + 2).array().min(0)).sum();
226
227 // Store nodal prices
228 Market.confirmed.price(tick, node_iter) = Market.bidded_price_map.bidded_price(0) + rep.lagbc[row_start];
229 Market.confirmed.price(tick, node_iter) = int(Market.confirmed.price(tick, node_iter)) + .5;
230 if(Market.confirmed.price(tick, node_iter) < Market.bidded_price_map.bidded_price(1)){
231 Market.confirmed.price(tick, node_iter) = Market.bidded_price_map.bidded_price(0);
232 }
233 else if(Market.confirmed.price(tick, node_iter) > Market.bidded_price_map.bidded_price(Market.price_intervals)){
234 Market.confirmed.price(tick, node_iter) = Market.bidded_price_map.bidded_price(Market.price_intervals + 1);
235 }
236
237 // Store ratio at nodes
238 for(int price_iter = 0; price_iter < Market.price_intervals + 2; ++ price_iter){
239 if(Market.bidded_price_map.bidded_price(price_iter) >= Market.confirmed.price(tick, node_iter) || price_iter == Market.price_intervals + 1){
240 if(sol[row_start + price_iter] >= 0.){
241 Market.confirmed.ratio_demand(node_iter) = std::min(Market.submitted_demand(price_iter, node_iter), Market.submitted_supply(price_iter, node_iter) - sol[row_start + price_iter]);
242 Market.confirmed.ratio_supply(node_iter) = Market.confirmed.ratio_demand(node_iter) + sol[row_start + price_iter];
243 Market.confirmed.ratio_demand(node_iter) /= Market.submitted_demand(price_iter, node_iter) + 1E-12;
244 Market.confirmed.ratio_supply(node_iter) /= Market.submitted_supply(price_iter, node_iter) + 1E-12;
245 }
246 else{
247 Market.confirmed.ratio_supply(node_iter) = std::min(Market.submitted_supply(price_iter, node_iter), Market.submitted_demand(price_iter, node_iter) + sol[row_start + price_iter]);
248 Market.confirmed.ratio_demand(node_iter) = Market.confirmed.ratio_supply(node_iter) - sol[row_start + price_iter];
249 Market.confirmed.ratio_demand(node_iter) /= Market.submitted_demand(price_iter, node_iter) + 1E-12;
250 Market.confirmed.ratio_supply(node_iter) /= Market.submitted_supply(price_iter, node_iter) + 1E-12;
251 }
252 break;
253 }
254 }
255 }
256
257 // Store scheduled power flow
258 Market.network.confirmed_power.row(tick) = sol_vec.tail(Market.network.num_edges);
259
260 std::cout << "DC Optimal Power Flow:\n";
261 switch(rep.terminationtype){
262 case -4:
263 std::cout << "LP problem is primal unbounded.\n";
264 break;
265 case -3:
266 std::cout << "LP problem is primal infeasible.\n";
267 break;
268 case 1:
269 std::cout << "LP problem is primal feasible.\n";
270 break;
271 }
272 std::cout << Market.confirmed.supply.row(tick).sum() << "\t" << Market.confirmed.demand.row(tick).sum() << "\n";
273 std::cout << sol_vec.segment(Market.network.num_vertice, Market.network.num_vertice).minCoeff() << " " << sol_vec.segment(Market.network.num_vertice, Market.network.num_vertice).maxCoeff() << " " << .5 * sol_vec.segment(Market.network.num_vertice, Market.network.num_vertice).array().abs().sum() << "\n";
274 std::cout << sol_vec.head(Market.network.num_vertice).minCoeff() << " " << sol_vec.head(Market.network.num_vertice).maxCoeff() << "\n";
275 std::cout << sol_vec.tail(Market.network.num_edges).minCoeff() << " " << sol_vec.tail(Market.network.num_edges).maxCoeff() << "\n\n";
276}

384void power_market::TSO_Market_Actual_Results_Get(int tick, market_inform &Market){
385 alglib::real_1d_array sol;
386 alglib::minlpreport rep;
387 alglib::minlpresults(Market.Problem, sol, rep);
388 Eigen::VectorXd sol_vec = Eigen::Map <Eigen::VectorXd> (sol.getcontent(), sol.length());
389
390 for(int node_iter = 0; node_iter < Market.network.num_vertice; ++ node_iter){
391 // Store power source / sink
392 int row_start = 2 * Market.network.num_vertice + node_iter * (Market.price_intervals + 2);
393 Market.actual.supply(tick, node_iter) = (sol_vec.segment(row_start, Market.price_intervals + 2).array().max(0)).sum();
394 Market.actual.demand(tick, node_iter) = -(sol_vec.segment(row_start, Market.price_intervals + 2).array().min(0)).sum();
395
396 // Store nodal prices
397 Market.actual.price(tick, node_iter) = Market.bidded_price_map.bidded_price(0) + rep.lagbc[row_start];
398 Market.actual.price(tick, node_iter) = int(Market.actual.price(tick, node_iter)) + .5;
399 if(Market.actual.price(tick, node_iter) < Market.bidded_price_map.bidded_price(1)){
400 Market.actual.price(tick, node_iter) = Market.bidded_price_map.bidded_price(0);
401 }
402 else if(Market.actual.price(tick, node_iter) > Market.bidded_price_map.bidded_price(Market.price_intervals)){
403 Market.actual.price(tick, node_iter) = Market.bidded_price_map.bidded_price(Market.price_intervals + 1);
404 }
405
406 // Store ratio at nodes
407 for(int price_iter = 0; price_iter < Market.price_intervals + 2; ++ price_iter){
408 if(Market.bidded_price_map.bidded_price(price_iter) >= Market.actual.price(tick, node_iter) || price_iter == Market.price_intervals + 1){
409 if(sol[row_start + price_iter] >= 0.){
410 Market.actual.ratio_demand(node_iter) = std::min(Market.submitted_demand(price_iter, node_iter), Market.submitted_supply(price_iter, node_iter) - sol[row_start + price_iter]);
411 Market.actual.ratio_supply(node_iter) = Market.actual.ratio_demand(node_iter) + sol[row_start + price_iter];
412 Market.actual.ratio_demand(node_iter) /= Market.submitted_demand(price_iter, node_iter) + 1E-12;
413 Market.actual.ratio_supply(node_iter) /= Market.submitted_supply(price_iter, node_iter) + 1E-12;
414 }
415 else{
416 Market.actual.ratio_supply(node_iter) = std::min(Market.submitted_supply(price_iter, node_iter), Market.submitted_demand(price_iter, node_iter) + sol[row_start + price_iter]);
417 Market.actual.ratio_demand(node_iter) = Market.actual.ratio_supply(node_iter) - sol[row_start + price_iter];
418 Market.actual.ratio_demand(node_iter) /= Market.submitted_demand(price_iter, node_iter) + 1E-12;
419 Market.actual.ratio_supply(node_iter) /= Market.submitted_supply(price_iter, node_iter) + 1E-12;
420 }
421 break;
422 }
423 }
424 //std::cout << node_iter << ":\t" << Market.actual_ratio_demand(node_iter) << "\t" << Market.actual_ratio_supply(node_iter) << "\n";
425 }
426 //std::cout << "\n";
427
428 // Store scheduled power flow
429 Market.network.actual_power.row(tick) = sol_vec.tail(Market.network.num_edges);
430
431 std::cout << "DC Optimal Power Flow:\n";
432 switch(rep.terminationtype){
433 case -4:
434 std::cout << "LP problem is primal unbounded.\n";
435 break;
436 case -3:
437 std::cout << "LP problem is primal infeasible.\n";
438 break;
439 case 1:
440 std::cout << "LP problem is primal feasible.\n";
441 break;
442 }
443 std::cout << Market.actual.supply.row(tick).sum() << "\t" << Market.actual.demand.row(tick).sum() << "\n";
444 std::cout << sol_vec.segment(Market.network.num_vertice, Market.network.num_vertice).minCoeff() << " " << sol_vec.segment(Market.network.num_vertice, Market.network.num_vertice).maxCoeff() << " " << .5 * sol_vec.segment(Market.network.num_vertice, Market.network.num_vertice).array().abs().sum() << "\n";
445 std::cout << sol_vec.head(Market.network.num_vertice).minCoeff() << " " << sol_vec.head(Market.network.num_vertice).maxCoeff() << "\n";
446 std::cout << sol_vec.tail(Market.network.num_edges).minCoeff() << " " << sol_vec.tail(Market.network.num_edges).maxCoeff() << "\n\n";
447}
