307 // Set the maps for levels of voltage and impedance base
308 void set_level_maps(edges &edges){
309 int voltage_min = edges.voltage_base.minCoeff();
310 int voltage_max = voltage_min;
311 int level_count = 0;
312 this->voltage_base_levels.insert(std::make_pair(voltage_max, level_count));
313 this->impedenace_base_levels.insert(std::make_pair(voltage_max, (double) this->phase_num * voltage_max * voltage_max / this->s_base));
314 this->power_limit.insert(std::make_pair(voltage_max, (double) this->line_trans * this->power_limit_trans * voltage_max));
315 //std::cout << voltage_base_levels[voltage_max] << "\t" << voltage_max << "\t" << impedenace_base_levels[voltage_max] << "\n";
316
317 std::vector <int> voltage_base_sorted(edges.voltage_base.data(), edges.voltage_base.data() + edges.voltage_base.size());
318 std::sort(voltage_base_sorted.begin(), voltage_base_sorted.end());
319 for(int edge_iter = 0; edge_iter < edges.voltage_base.size(); ++ edge_iter){
320 if(voltage_base_sorted[edge_iter] > voltage_max){
321 voltage_max = voltage_base_sorted[edge_iter];
322 level_count += 1;
323 this->voltage_base_levels.insert(std::make_pair(voltage_max, level_count));
324 this->impedenace_base_levels.insert(std::make_pair(voltage_max, (double) this->phase_num * voltage_max * voltage_max / this->s_base));
325 this->power_limit.insert(std::make_pair(voltage_max, (double) this->line_trans * this->power_limit_trans * voltage_max));
326 //std::cout << voltage_base_levels[voltage_max] << "\t" << voltage_max << "\t" << impedenace_base_levels[voltage_max] << "\n";
327 }
328 }
