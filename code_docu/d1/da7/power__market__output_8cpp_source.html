8 void Simplified_network_print(power_market::market_inform &Market, std::string name){
9 int edge_num = Market.network.num_edges;
10 std::string fout_name;
11
12 // Set edge names
13 std::vector <std::string> col_names(3);
14 col_names[0] = "from";
15 col_names[1] = "to";
16 col_names[2] = "power_capacity";
17
18 Eigen::MatrixXd output(edge_num, 3);
19 for(int edge_iter = 0; edge_iter < edge_num; ++ edge_iter){
20 output(edge_iter, 0) = Market.network.incidence[edge_iter](0);
21 output(edge_iter, 1) = Market.network.incidence[edge_iter](1);
22 output(edge_iter, 2) = Market.network.power_constraint(edge_iter, 1);
23 }
