20 int DSO_node_num = Power_network_inform.DSO_cluster[DSO_iter].nodes_ID.size();

22 DSO_Markets[DSO_iter].num_zone = DSO_node_num;
23 DSO_Markets[DSO_iter].network.num_edges = 0;
24 //std::cout << DSO_iter << ":\t" << DSO_Markets[DSO_iter].num_zone << "\t" << DSO_Markets[DSO_iter].network.num_edges << "\n";
25 continue;
26 }
27
28 DSO_Markets[DSO_iter].num_zone = DSO_point_num + DSO_node_num;
29 DSO_Markets[DSO_iter].time_intervals = Time;
30 parameters::bidded_price(DSO_Markets[DSO_iter].bidded_price_map);
31
32 // Set compact incidence matrix and edge admittance matrix
33 double tol = 1.;
34 DSO_Markets[DSO_iter].network.incidence.reserve(DSO_Markets[DSO_iter].network.num_vertice * DSO_Markets[DSO_iter].network.num_vertice);
35 DSO_Markets[DSO_iter].network.admittance.reserve(DSO_Markets[DSO_iter].network.num_vertice * DSO_Markets[DSO_iter].network.num_vertice);

37 power_limit.reserve(DSO_Markets[DSO_iter].network.num_vertice * DSO_Markets[DSO_iter].network.num_vertice);

39 // Calculate edge admittance
40 // Use fractional Laplacian here :
41 // y(x_1, x_2) = N / 4 / pi / z / |x_1 - x_2|^(2 + d), where
42 // y(x_1, x_2): admittance between x_1 and x_2
43 // N: mean line density at the neighborhood of x_1 containing also x_2
44 // z: per length impedence of power lines
45 // d: fractional dimension of the power line distribution
46 //= pow(Power_network_inform.tech_parameters.voltage_cutoff_distr, 2.) / Power_network_inform.tech_parameters.s_base;
47 //double z_base_high = pow(Power_network_inform.tech_parameters.voltage_cutoff_connection, 2.) / Power_network_inform.tech_parameters.s_base;
48 double partition_func = 0.;
49 DSO_Markets[DSO_iter].network.num_vertice = DSO_Markets[DSO_iter].num_zone;
50 Eigen::MatrixXd num_line = Eigen::MatrixXd::Ones(DSO_point_num, DSO_point_num);
51 Eigen::MatrixXd distance = Eigen::MatrixXd::Zero(DSO_point_num, DSO_point_num);
52
53 // Connection between points
54 for(int row_iter = 0; row_iter < DSO_point_num - 1; ++ row_iter){
55 for(int col_iter = row_iter + 1; col_iter < DSO_point_num ; ++ col_iter){
56 int point_ID_1 = Power_network_inform.DSO_cluster[DSO_iter].points_ID[row_iter];
57 int point_ID_2 = Power_network_inform.DSO_cluster[DSO_iter].points_ID[col_iter];
58
59 distance(row_iter, col_iter) = Power_network_inform.points.distance(point_ID_1, point_ID_2);
60 num_line(row_iter, col_iter) = 1. / pow(distance(row_iter, col_iter) * 1E-3, 1. + Power_network_inform.tech_parameters.fraction_dim_distr);
61 partition_func += num_line(row_iter, col_iter);
62 }
63 }
64 num_line /= partition_func;
65 num_line *= Power_network_inform.tech_parameters.line_density_distr * DSO_point_num;
66
67 for(int row_iter = 0; row_iter < DSO_point_num - 1; ++ row_iter){
68 for(int col_iter = row_iter + 1; col_iter < DSO_point_num ; ++ col_iter){
69 // Series admittance
70 double y_series = 1.;
71 y_series /= distance(row_iter, col_iter);
72 y_series /= Power_network_inform.tech_parameters.z_distr_series.imag();
73 y_series *= z_base_low;
74 y_series *= num_line(row_iter, col_iter);
75
76 DSO_Markets[DSO_iter].network.incidence.push_back(Eigen::Vector2i(row_iter, col_iter));
77 DSO_Markets[DSO_iter].network.admittance.push_back(y_series);
78 power_limit.push_back(power_limit_distr * Power_network_inform.tech_parameters.voltage_cutoff_distr * num_line(row_iter, col_iter));
79 }
80 }
81
82 // Connection between nodes and points
83 for(int point_iter = 0; point_iter < DSO_point_num; ++ point_iter){
84 int point_ID = Power_network_inform.DSO_cluster[DSO_iter].points_ID[point_iter];
85 int node_ID = Power_network_inform.points.node(point_ID);
86 int in_cluster_node_ID = Power_network_inform.nodes.in_cluster_ID(node_ID);
87 double line_density_conn = Power_network_inform.tech_parameters.line_density_conn;
88
89 Eigen::Vector2d point_coor = Eigen::Vector2d(Power_network_inform.points.lon(point_ID), Power_network_inform.points.lat(point_ID));
90 Eigen::Vector2d node_coor = Eigen::Vector2d(Power_network_inform.nodes.lon(node_ID), Power_network_inform.nodes.lat(node_ID));
91 point_coor *= pi / 180.;
92 node_coor *= pi / 180.;
93 double distance_temp = spatial_field::geodist(point_coor, node_coor);
94
95 // Series admittance
96 double y_series = 1.;
97 y_series /= distance_temp;
98 y_series /= Power_network_inform.tech_parameters.z_conn_series.imag();
99 y_series *= z_base_high;
100 y_series *= line_density_conn;
101
102 DSO_Markets[DSO_iter].network.incidence.push_back(Eigen::Vector2i(point_iter, DSO_point_num + in_cluster_node_ID));
103 DSO_Markets[DSO_iter].network.admittance.push_back(y_series);
104 power_limit.push_back(power_limit_conn * Power_network_inform.tech_parameters.voltage_cutoff_conn * line_density_conn);
105 }
106 DSO_Markets[DSO_iter].network.num_edges = DSO_Markets[DSO_iter].network.incidence.size();
107 //std::cout << DSO_iter << ":\t" << DSO_Markets[DSO_iter].num_zone << "\t" << DSO_Markets[DSO_iter].network.num_edges << "\n";
108
109 // Set voltage and power constraints at each edge
110 DSO_Markets[DSO_iter].network.voltage_constraint = Eigen::MatrixXd::Ones(DSO_Markets[DSO_iter].network.num_vertice, 2);
111 DSO_Markets[DSO_iter].network.voltage_constraint.col(0) *= -Power_network_inform.tech_parameters.theta_distr_limit;
112 DSO_Markets[DSO_iter].network.voltage_constraint.col(1) *= Power_network_inform.tech_parameters.theta_distr_limit;
113 DSO_Markets[DSO_iter].network.power_constraint = Eigen::MatrixXd (DSO_Markets[DSO_iter].network.num_edges, 2);
114 DSO_Markets[DSO_iter].network.power_constraint.col(1) = Eigen::Map <Eigen::VectorXd> (power_limit.data(), power_limit.size());
115 DSO_Markets[DSO_iter].network.power_constraint.col(1) /= Power_network_inform.tech_parameters.s_base;
116 DSO_Markets[DSO_iter].network.power_constraint.col(0) = -DSO_Markets[DSO_iter].network.power_constraint.col(1);
117
118 // Initialization of process variables
119 power_market::Market_Initialization(DSO_Markets[DSO_iter]);
120
121 // Initialization of output variables
122 DSO_Markets[DSO_iter].confirmed.supply = Eigen::MatrixXd::Zero(Time, DSO_Markets[DSO_iter].num_zone);
123 DSO_Markets[DSO_iter].confirmed.demand = Eigen::MatrixXd::Zero(Time, DSO_Markets[DSO_iter].num_zone);
124 DSO_Markets[DSO_iter].confirmed.price = Eigen::MatrixXd::Zero(Time, DSO_Markets[DSO_iter].num_zone);
125 DSO_Markets[DSO_iter].confirmed.ratio_supply = Eigen::VectorXd::Zero(DSO_Markets[DSO_iter].num_zone);

271void power_market::DSO_Market_Results_Get(int tick, market_inform &Market, power_network::DSO_cluster &DSO_cluster, bool supply){
272 alglib::real_1d_array sol;
273 alglib::minlpreport rep;
274 alglib::minlpresults(Market.Problem, sol, rep);
275 Eigen::VectorXd sol_vec = Eigen::Map <Eigen::VectorXd> (sol.getcontent(), sol.length());
276
277 if(supply){
278 // Store filtered supply bids at spatial points
279 for(int point_iter = 0; point_iter < DSO_cluster.points_ID.size(); ++ point_iter){
280 // Store power sink
281 int row_start = 2 * Market.network.num_vertice + point_iter * (Market.price_intervals + 2);
282 Market.confirmed.supply(tick, point_iter) = (sol_vec.segment(row_start, Market.price_intervals + 2).array().max(0)).sum();
283
284 // Store nodal prices
285 Market.confirmed.price(tick, point_iter) = Market.bidded_price_map.bidded_price(1) + rep.lagbc[row_start + 1];
286 Market.confirmed.price(tick, point_iter) = int(Market.confirmed.price(tick, point_iter)) + .5;
287 if(Market.confirmed.price(tick, point_iter) < Market.bidded_price_map.bidded_price(1)){
288 Market.confirmed.price(tick, point_iter) = Market.bidded_price_map.bidded_price(0);
289 }
290 else if(Market.confirmed.price(tick, point_iter) > Market.bidded_price_map.bidded_price(Market.price_intervals)){
291 Market.confirmed.price(tick, point_iter) = Market.bidded_price_map.bidded_price(Market.price_intervals + 1);
292 }
293
294 // Store ratio at nodes
295 for(int price_iter = 0; price_iter < Market.price_intervals + 2; ++ price_iter){
296 if(Market.bidded_price_map.bidded_price(price_iter) >= Market.confirmed.price(tick, point_iter) || price_iter == Market.price_intervals + 1){
297 if(sol[row_start + price_iter] >= 0.){
298 Market.confirmed.ratio_demand(point_iter) = std::min(Market.submitted_demand(price_iter, point_iter), Market.submitted_supply(price_iter, point_iter) - sol[row_start + price_iter]);
299 Market.confirmed.ratio_supply(point_iter) = Market.confirmed.ratio_demand(point_iter) + sol[row_start + price_iter];
300 Market.confirmed.ratio_supply(point_iter) /= Market.submitted_supply(price_iter, point_iter) + 1E-12;
301 }
302 else{
303 Market.confirmed.ratio_supply(point_iter) = std::min(Market.submitted_supply(price_iter, point_iter), Market.submitted_demand(price_iter, point_iter) + sol[row_start + price_iter]);
304 Market.confirmed.ratio_supply(point_iter) /= Market.submitted_supply(price_iter, point_iter) + 1E-12;
305 }
306 break;
307 }
308 }
309 }
310 }
311 else{
312 // Store filtered demand bids at spatial points
313 for(int point_iter = 0; point_iter < DSO_cluster.points_ID.size(); ++ point_iter){
314 // Store power sink
315 int row_start = 2 * Market.network.num_vertice + point_iter * (Market.price_intervals + 2);
316 Market.confirmed.demand(tick, point_iter) = -(sol_vec.segment(row_start, Market.price_intervals + 2).array().min(0)).sum();
317
318 // Store nodal prices
319 Market.confirmed.price(tick, point_iter) = Market.bidded_price_map.bidded_price(0) + rep.lagbc[row_start];
320 Market.confirmed.price(tick, point_iter) = int(Market.confirmed.price(tick, point_iter)) + .5;
321 if(Market.confirmed.price(tick, point_iter) < Market.bidded_price_map.bidded_price(1)){
322 Market.confirmed.price(tick, point_iter) = Market.bidded_price_map.bidded_price(0);
323 }
324 else if(Market.confirmed.price(tick, point_iter) > Market.bidded_price_map.bidded_price(Market.price_intervals)){
325 Market.confirmed.price(tick, point_iter) = Market.bidded_price_map.bidded_price(Market.price_intervals + 1);
326 }
327
328 // Store ratio at nodes
329 for(int price_iter = 0; price_iter < Market.price_intervals + 2; ++ price_iter){
330 if(Market.bidded_price_map.bidded_price(price_iter) >= Market.confirmed.price(tick, point_iter) || price_iter == Market.price_intervals + 1){
331 if(sol[row_start + price_iter] >= 0.){
332 Market.confirmed.ratio_demand(point_iter) = std::min(Market.submitted_demand(price_iter, point_iter), Market.submitted_supply(price_iter, point_iter) - sol[row_start + price_iter]);
333 Market.confirmed.ratio_demand(point_iter) /= Market.submitted_demand(price_iter, point_iter) + 1E-12;
334 }
335 else{
336 Market.confirmed.ratio_supply(point_iter) = std::min(Market.submitted_supply(price_iter, point_iter), Market.submitted_demand(price_iter, point_iter) + sol[row_start + price_iter]);
337 Market.confirmed.ratio_demand(point_iter) = Market.confirmed.ratio_supply(point_iter) - sol[row_start + price_iter];
338 Market.confirmed.ratio_demand(point_iter) /= Market.submitted_demand(price_iter, point_iter) + 1E-12;
339 }
340 break;
341 }
342 }
343 }
