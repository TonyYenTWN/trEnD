7void power_market::parameters::bidded_price(price_ID_bimap &obj){
8 int price_intervals = price_interval();
9 // Range of lowest and highest possible bidding prices.
10 Eigen::Vector2d price_range_inflex = Eigen::Vector2d(-500., 3000.);
11 // Range of bidding prices for flexible supply and demand in the model.
12 Eigen::Vector2d price_range_flex = Eigen::Vector2d(-100., 500.);
13
14 Eigen::VectorXd bidded_price = Eigen::VectorXd(price_intervals + 2);
15 bidded_price(0) = price_range_inflex(0);
16 bidded_price.array().tail(1) = price_range_inflex(1);
17 bidded_price.segment(1, price_intervals) = Eigen::VectorXd::LinSpaced(price_intervals, price_range_flex(0) + .5 * (price_range_flex(1) - price_range_flex(0)) / price_intervals, price_range_flex(1) - .5 * (price_range_flex(1) - price_range_flex(0)) / price_intervals);
18
19 std::map <double, int> price_ID;
20 for(int price_iter = 0; price_iter < bidded_price.size(); ++ price_iter){
21 price_ID.insert(std::pair <double, int> (bidded_price(price_iter), price_iter));
22 }
23
24 obj.bidded_price = bidded_price;
25 obj.price_ID = price_ID;
26}

69void power_market::Flow_Based_Market_LP_Set(market_inform &Market){
70 // -------------------------------------------------------------------------------
71 // LP Solver initialization for flow-based market optimization
72 // Warm-up once and reuse for the rest of time slices
73 // Variables are sorted as {V}, {S}, {S}_0, {S}_1, {S}_2, ..., {I}
74 // {S}_i is the set of source / sink at node #i at different price levels
75 // -------------------------------------------------------------------------------
76
77 // -------------------------------------------------------------------------------
78 // Set matrix for general constraints
79 // -------------------------------------------------------------------------------
80 // Construct node admittance matrix
81 std::vector <Eigen::TripletXd> Y_n_trip;
82 Y_n_trip.reserve(2 * Market.network.num_edges + Market.network.num_vertice);
83 Eigen::SparseMatrix <double, Eigen::RowMajor> Y_n(Market.network.num_vertice, Market.network.num_vertice);
84 Eigen::VectorXd Y_n_diag = Eigen::VectorXd::Zero(Market.network.num_vertice);
85 Eigen::VectorXpd Connection_num = Eigen::VectorXpd::Ones(Market.network.num_vertice);
86 for(int edge_iter = 0; edge_iter < Market.network.num_edges; ++ edge_iter){
87 double y_edge = Market.network.admittance[edge_iter];
88
89 // Equality constraints of voltage - source / sink at the nodes, off-diagonal terms
90 Y_n_trip.push_back(Eigen::TripletXd(Market.network.incidence[edge_iter](0), Market.network.incidence[edge_iter](1), -y_edge));
91 Y_n_trip.push_back(Eigen::TripletXd(Market.network.incidence[edge_iter](1), Market.network.incidence[edge_iter](0), -y_edge));
92 Connection_num(Market.network.incidence[edge_iter](0)) += 1;
93 Connection_num(Market.network.incidence[edge_iter](1)) += 1;
94
95 // Equality constraints of voltage - source / sink at the nodes, diagonal terms
96 Y_n_diag(Market.network.incidence[edge_iter](0)) += y_edge;
97 Y_n_diag(Market.network.incidence[edge_iter](1)) += y_edge;
98 }
99 for(int node_iter = 0; node_iter < Market.network.num_vertice; ++ node_iter){
100 // Equality constraints of voltage - source / sink at the nodes, summed diagonal terms
101 Y_n_trip.push_back(Eigen::TripletXd(node_iter, node_iter, Y_n_diag(node_iter)));
102 }
103 Y_n.setFromTriplets(Y_n_trip.begin(), Y_n_trip.end());
104
105 // Generate sparse matrix for general (equality) constraints of voltage, power flow, and source / sink summation
106 int constrant_num = 2 * Market.network.num_vertice + Market.network.num_edges;
107 int variable_num = Market.network.num_vertice * (Market.price_intervals + 4) + Market.network.num_edges;
108 Eigen::VectorXpd non_zero_num(constrant_num);
109 non_zero_num << Connection_num + Eigen::VectorXpd::Ones(Market.network.num_vertice), Eigen::VectorXpd::Constant(Market.network.num_edges, 3), Eigen::VectorXpd::Constant(Market.network.num_vertice, Market.price_intervals + 3);
110 alglib::integer_1d_array row_sizes_general;
111 row_sizes_general.setcontent(non_zero_num.size(), non_zero_num.data());
112 alglib::sparsematrix constraint_general;
113 alglib::sparsecreatecrs(constrant_num, variable_num, row_sizes_general, constraint_general);
114 // Rows for voltage - source / sink equalities
115 for(int row_iter = 0; row_iter < Y_n.outerSize(); ++ row_iter){
116 for(Eigen::SparseMatrix<double, Eigen::RowMajor>::InnerIterator inner_iter(Y_n, row_iter); inner_iter; ++ inner_iter){
117 alglib::sparseset(constraint_general, inner_iter.row(), inner_iter.col(), inner_iter.value());
118 }
119 // Update the columns right to the node admittance block
120 alglib::sparseset(constraint_general, row_iter, Market.network.num_vertice + row_iter, -1.);
121 }
122 // Rows for voltage - power flow equalities
123 for(int edge_iter = 0; edge_iter < Market.network.num_edges; ++ edge_iter){
124 double y_edge = Market.network.admittance[edge_iter];
125
126 if(Market.network.incidence[edge_iter](0) < Market.network.incidence[edge_iter](1)){
127 alglib::sparseset(constraint_general, Y_n.rows() + edge_iter, Market.network.incidence[edge_iter](0), y_edge);
128 alglib::sparseset(constraint_general, Y_n.rows() + edge_iter, Market.network.incidence[edge_iter](1), -y_edge);
129 }
130 else{
131 alglib::sparseset(constraint_general, Y_n.rows() + edge_iter, Market.network.incidence[edge_iter](0), -y_edge);
132 alglib::sparseset(constraint_general, Y_n.rows() + edge_iter, Market.network.incidence[edge_iter](1), y_edge);
133 }
134 alglib::sparseset(constraint_general, Y_n.rows() + edge_iter, Market.network.num_vertice * (Market.price_intervals + 4) + edge_iter, -1.);
135 }
136
137 // Rows for source / sink summation at each node
138 for(int node_iter = 0; node_iter < Y_n.rows(); ++ node_iter){
139 int row_ID = Y_n.rows() + Market.network.num_edges + node_iter;
140 alglib::sparseset(constraint_general, row_ID, Y_n.rows() + node_iter, 1.);
141 int col_start = 2 * Y_n.rows() + node_iter * (Market.price_intervals + 2);
142 int col_end = 2 * Y_n.rows() + (node_iter + 1) * (Market.price_intervals + 2) - 1;
143 for(int col_iter = col_start; col_iter <= col_end; ++ col_iter){
144 alglib::sparseset(constraint_general, row_ID, col_iter, -1.);
145 }
146 }
147
148 // Check if the sparse matrix is correct
149// std::cout << Y_n << "\n\n";
150// double value;
151// for(int row_iter = 0; row_iter < non_zero_num.size(); ++ row_iter){
152// for(int col_iter = 0; col_iter < 2 * Y_n.cols() + 3; ++ col_iter){
153// value = sparseget(constraint_general, row_iter, col_iter);
154// std::cout << value << "\t";
155// }
156// std::cout << "\t";
157// for(int col_iter = Market.network.num_vertice * (Market.price_intervals + 4); col_iter < Market.network.num_vertice * (Market.price_intervals + 4) + 1; ++ col_iter){
158// value = sparseget(constraint_general, row_iter, col_iter);
159// std::cout << value << "\t";
160// }
161// std::cout << "\n";
162// }
163
164 // -------------------------------------------------------------------------------
165 // Set bounds for general and box constraints
166 // -------------------------------------------------------------------------------
167 Eigen::MatrixXd bound_general = Eigen::MatrixXd::Zero(constrant_num, 2);
168 Eigen::MatrixXd bound_box = Eigen::MatrixXd::Zero(variable_num, 2);
169 bound_box.topRows(Market.network.num_vertice) = Market.network.voltage_constraint;
170 bound_box.middleRows(Market.network.num_vertice, Market.network.num_vertice).col(0) = Eigen::VectorXd::Constant(Market.network.num_vertice, -std::numeric_limits<double>::infinity());
171 bound_box.middleRows(Market.network.num_vertice, Market.network.num_vertice).col(1) = Eigen::VectorXd::Constant(Market.network.num_vertice, std::numeric_limits<double>::infinity());
172 bound_box.bottomRows(Market.network.num_edges) = Market.network.power_constraint;
173
174 // Bounds of general constraints
175 alglib::real_1d_array lb_general;
176 alglib::real_1d_array ub_general;
177 lb_general.setcontent(bound_general.rows(), bound_general.col(0).data());
178 ub_general.setcontent(bound_general.rows(), bound_general.col(1).data());
179
180 // -------------------------------------------------------------------------------
181 // Set scale of variables
182 // -------------------------------------------------------------------------------
183 Eigen::VectorXd scale_vec = Eigen::VectorXd::Ones(variable_num);
184 scale_vec.head(Market.network.num_vertice) = bound_box.col(1).head(Market.network.num_vertice) - bound_box.col(0).head(Market.network.num_vertice);
185 scale_vec.tail(Market.network.num_edges) = bound_box.col(1).tail(Market.network.num_edges) - bound_box.col(0).tail(Market.network.num_edges);
186 alglib::real_1d_array scale;
187 scale.setcontent(scale_vec.size(), scale_vec.data());
188
189 // -------------------------------------------------------------------------------
190 // Set objective coefficients of variables
191 // -------------------------------------------------------------------------------
192 Eigen::VectorXd obj_vec = Eigen::VectorXd::Zero(variable_num);
193 for(int node_iter = 0; node_iter < Market.network.num_vertice; ++ node_iter){
194 int row_start = 2 * Market.network.num_vertice + node_iter * (Market.price_intervals + 2);
195 obj_vec.segment(row_start, Market.price_intervals + 2) = Market.bidded_price_map.bidded_price;
196 }
197 alglib::real_1d_array obj_coeff;
198 obj_coeff.setcontent(obj_vec.size(), obj_vec.data());
199
200 // -------------------------------------------------------------------------------
201 // Set the LP problem object
202 // -------------------------------------------------------------------------------
203 alglib::minlpcreate(variable_num, Market.Problem);
204 alglib::minlpsetcost(Market.Problem, obj_coeff);
205 alglib::minlpsetlc2(Market.Problem, constraint_general, lb_general, ub_general, constrant_num);
206 alglib::minlpsetscale(Market.Problem, scale);
207 alglib::minlpsetalgodss(Market.Problem, 0.);
208}

210void power_market::Flow_Based_Market_Optimization(market_inform &Market){
211 // -------------------------------------------------------------------------------
212 // Update bounds for box constraints
213 // -------------------------------------------------------------------------------
214 int variable_num = Market.network.num_vertice * (Market.price_intervals + 4) + Market.network.num_edges;
215 Eigen::MatrixXd bound_box(variable_num, 2);
216 bound_box.topRows(Market.network.num_vertice) = Market.network.voltage_constraint;
217 bound_box.middleRows(Market.network.num_vertice, Market.network.num_vertice).col(0) = Eigen::VectorXd::Constant(Market.network.num_vertice, -std::numeric_limits<double>::infinity());
218 bound_box.middleRows(Market.network.num_vertice, Market.network.num_vertice).col(1) = Eigen::VectorXd::Constant(Market.network.num_vertice, std::numeric_limits<double>::infinity());
219 for(int node_iter = 0; node_iter < Market.network.num_vertice; ++ node_iter){
220 int row_start = 2 * Market.network.num_vertice + node_iter * (Market.price_intervals + 2);
221 bound_box.middleRows(row_start, Market.price_intervals + 2).col(0) = -Market.submitted_demand.col(node_iter);
222 bound_box.middleRows(row_start, Market.price_intervals + 2).col(1) = Market.submitted_supply.col(node_iter);
223 }
224 bound_box.bottomRows(Market.network.num_edges) = Market.network.power_constraint;
225
226 // Bounds of box constraints
227 alglib::real_1d_array lb_box;
228 alglib::real_1d_array ub_box;
229 lb_box.setcontent(bound_box.rows(), bound_box.col(0).data());
230 ub_box.setcontent(bound_box.rows(), bound_box.col(1).data());
231 alglib::minlpsetbc(Market.Problem, lb_box, ub_box);
232
233 // -------------------------------------------------------------------------------
234 // Set objective coefficients of variables
235 // -------------------------------------------------------------------------------
236 Eigen::VectorXd obj_vec = Eigen::VectorXd::Zero(variable_num);
237 for(int node_iter = 0; node_iter < Market.network.num_vertice; ++ node_iter){
238 int row_start = 2 * Market.network.num_vertice + node_iter * (Market.price_intervals + 2);
239 obj_vec.segment(row_start, Market.price_intervals + 2) = Market.bidded_price_map.bidded_price;
240 obj_vec.segment(row_start, Market.price_intervals + 2) -= Eigen::VectorXd::Constant(Market.reference_price(node_iter), Market.price_intervals + 2);
241 }
242 alglib::real_1d_array obj_coeff;
243 obj_coeff.setcontent(obj_vec.size(), obj_vec.data());
244 alglib::minlpsetcost(Market.Problem, obj_coeff);
245
246 // -------------------------------------------------------------------------------
247 // Solve the problem
248 // -------------------------------------------------------------------------------
249 alglib::minlpoptimize(Market.Problem);
250}
