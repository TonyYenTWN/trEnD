59void power_market::International_Market_Set(market_inform &International_Market, power_network::network_inform &Power_network_inform, int Time, fin_market fin_market){
60 // Input Parameters of international market
61 International_Market.num_zone = Power_network_inform.cbt.bz_names.size();
62 International_Market.cross_border_zone_start = Power_network_inform.points.bidding_zone.maxCoeff() + 1;
63 International_Market.time_intervals = Time;
64 International_Market.zone_names = Power_network_inform.cbt.bz_names;
65 parameters::bidded_price(International_Market.bidded_price_map);
66 International_Market.network.num_vertice = International_Market.num_zone;
67 International_Market.network.line_capacity_matrix = Power_network_inform.cbt.flow_constraint;
68
69 // Construct incidence vector matrix
70 International_Market.network.incidence.reserve(International_Market.network.num_vertice * International_Market.network.num_vertice);
71 for(int row_iter = 0; row_iter < International_Market.network.num_vertice - 1; ++ row_iter){
72 for(int col_iter = row_iter + 1; col_iter < International_Market.network.num_vertice; ++ col_iter){
73 bool add_flag = 1 - (International_Market.network.line_capacity_matrix(row_iter, col_iter) < 0.) * (International_Market.network.line_capacity_matrix(col_iter, row_iter) < 0.);
74 if(add_flag){
75 International_Market.network.incidence.push_back(Eigen::Vector2i(row_iter, col_iter));
76 }
77 }
78 }
79 International_Market.network.line_capacity_matrix = Eigen::MatrixXd::Ones(International_Market.num_zone, International_Market.num_zone).array() * International_Market.network.line_capacity_matrix.array().max(0);
80
81 // Construct power constraint matrix
82 International_Market.network.num_edges = International_Market.network.incidence.size();
83 International_Market.network.power_constraint = Eigen::MatrixXd(International_Market.network.num_edges, 2);
84 for(int edge_iter = 0; edge_iter < International_Market.network.num_edges; ++ edge_iter){
85 int row_ID = International_Market.network.incidence[edge_iter](0);
86 int col_ID = International_Market.network.incidence[edge_iter](1);
87 International_Market.network.power_constraint(edge_iter, 0) = International_Market.network.line_capacity_matrix(row_ID, col_ID);
88 International_Market.network.power_constraint(edge_iter, 1) = International_Market.network.line_capacity_matrix(col_ID, row_ID);
89 }
90
91 // Quantity density at each price
92 // Read inferred merit order curve data
93 auto moc_dim = basic::get_file_dim(fin_market.moc);
94 Eigen::MatrixXd merit_order_curve_q = basic::read_file(moc_dim[0], moc_dim[1], fin_market.moc);
95 Eigen::MatrixXd diff_merit_order_curve_q = merit_order_curve_q.bottomRows(moc_dim[0] - 1) - merit_order_curve_q.topRows(moc_dim[0] - 1);
96 International_Market.merit_order_curve = merit_order_curve_q;
97 International_Market.merit_order_curve.bottomRows(moc_dim[0] - 1) = diff_merit_order_curve_q;
98 International_Market.merit_order_curve = Eigen::MatrixXd::Ones(moc_dim[0], moc_dim[1]).array() * International_Market.merit_order_curve.array().max(0);
99
100 // Default (residual) demand time series
101 // Read default demand data
102 auto demand_ts_dim = basic::get_file_dim(fin_market.demand);
103 International_Market.demand_default = basic::read_file(demand_ts_dim[0], demand_ts_dim[1], fin_market.demand, 1);
104
105 // Read cbt data
106 auto cbt_ts_dim = basic::get_file_dim(fin_market.cbt);
107 auto cbt_ts = basic::read_file(cbt_ts_dim[0], cbt_ts_dim[1], fin_market.cbt, 1);
108
109 // Adjusted demand profile after cbt is taken out
110 for(int edge_iter = 0; edge_iter < International_Market.network.incidence.size(); ++ edge_iter){
111 int start_zone_ID = International_Market.network.incidence[edge_iter](0);
112 int end_zone_ID = International_Market.network.incidence[edge_iter](1);
113 int start_edge_ID = 2 * edge_iter;
114 int end_edge_ID = 2 * edge_iter + 1;
115 International_Market.demand_default.col(start_zone_ID) -= cbt_ts.col(start_edge_ID);
116 International_Market.demand_default.col(start_zone_ID) += cbt_ts.col(end_edge_ID);
117 International_Market.demand_default.col(end_zone_ID) += cbt_ts.col(start_edge_ID);
118 International_Market.demand_default.col(end_zone_ID) -= cbt_ts.col(end_edge_ID);
119 }
120
121 // Calculate default residual demand profile from subtracting VRE for nations on the boundary
122 auto solar_ts_dim = basic::get_file_dim(fin_market.solar);
123 auto solar_ts = basic::read_file(solar_ts_dim[0], solar_ts_dim[1], fin_market.solar, 1);
124 auto wind_on_ts_dim = basic::get_file_dim(fin_market.wind_on);
125 auto wind_on_ts = basic::read_file(wind_on_ts_dim[0], wind_on_ts_dim[1], fin_market.wind_on, 1);
126 auto wind_off_ts_dim = basic::get_file_dim(fin_market.wind_off);
127 auto wind_off_ts = basic::read_file(wind_off_ts_dim[0], wind_off_ts_dim[1], fin_market.wind_off, 1);
128 int boundary_num = International_Market.num_zone - International_Market.cross_border_zone_start;
129 International_Market.demand_default.rightCols(boundary_num) -= solar_ts.rightCols(boundary_num) + wind_on_ts.rightCols(boundary_num) + wind_off_ts.rightCols(boundary_num);
130
131 // Initialization of process variables
132 power_market::Market_Initialization(International_Market);
133
134 // Initialization of output variables
135 International_Market.confirmed.supply = Eigen::MatrixXd::Zero(Time, International_Market.num_zone);
136 International_Market.confirmed.demand = Eigen::MatrixXd::Zero(Time, International_Market.num_zone);
137 International_Market.confirmed.price = Eigen::MatrixXd(Time, International_Market.num_zone);
138 International_Market.confirmed.ratio_supply = Eigen::VectorXd::Zero(International_Market.num_zone);
139 International_Market.confirmed.ratio_demand = Eigen::VectorXd::Zero(International_Market.num_zone);
140 International_Market.network.confirmed_power = Eigen::MatrixXd::Zero(Time, International_Market.network.num_edges);
141 International_Market.redispatch.price_demand = Eigen::MatrixXd::Zero(Time, International_Market.num_zone);
142 International_Market.redispatch.price_supply = Eigen::MatrixXd::Zero(Time, International_Market.num_zone);
143 Operation_Initialization(International_Market);
144
145 // Update alglib object for optimization
146 // -------------------------------------------------------------------------------
147 // Set matrix for general constraints
148 // -------------------------------------------------------------------------------
149 // Variables are sorted as {F}, {S}, {S}_0, {S}_1, {S}_2, ..., {I}
150 // {F} are power flows across the edges
151 // {S}_i is the set of source / sink at node #i at different price levels
152 std::vector <Eigen::TripletXd> Constraint_trip;
153 Constraint_trip.reserve(International_Market.network.num_vertice * (International_Market.price_intervals + 4) + 2 * International_Market.network.num_edges);
154 Eigen::VectorXpd Connection_num = Eigen::VectorXpd::Zero(International_Market.network.num_vertice);
155 // Constraint for energy conservation
156 for(int node_iter = 0; node_iter < International_Market.network.num_vertice; ++ node_iter){
157 int col_ID = International_Market.network.num_edges + node_iter;
158 Constraint_trip.push_back(Eigen::TripletXd(node_iter, col_ID, 1.));
159 }
160 for(int edge_iter = 0; edge_iter < International_Market.network.num_edges; ++ edge_iter){
161 int from_ID = International_Market.network.incidence[edge_iter](0);
162 int to_ID = International_Market.network.incidence[edge_iter](1);
163 Constraint_trip.push_back(Eigen::TripletXd(from_ID, edge_iter, -1.));
164 Constraint_trip.push_back(Eigen::TripletXd(to_ID, edge_iter, 1.));
165 Connection_num(from_ID) += 1;
166 Connection_num(to_ID) += 1;
167 }
168
169 // Rows for source / sink summation at each node
170 for(int node_iter = 0; node_iter < International_Market.network.num_vertice; ++ node_iter){
171 int row_ID = International_Market.network.num_vertice + node_iter;
172 int col_ID = International_Market.network.num_edges + node_iter;
173 Constraint_trip.push_back(Eigen::TripletXd(row_ID, col_ID, 1.));
174
175 int col_start = International_Market.network.num_edges + International_Market.network.num_vertice + node_iter * (International_Market.price_intervals + 2);
176 int col_end = col_start + International_Market.price_intervals + 1;
177 for(int col_iter = col_start; col_iter <= col_end; ++ col_iter){
178 Constraint_trip.push_back(Eigen::TripletXd(row_ID, col_iter, -1.));
179 }
180 }
181
182 int constrant_num = 2 * International_Market.network.num_vertice;
183 int variable_num = International_Market.network.num_edges + International_Market.network.num_vertice * (International_Market.price_intervals + 3);
184 Eigen::SparseMatrix <double, Eigen::RowMajor> Constraint(constrant_num, variable_num);
185 Constraint.setFromTriplets(Constraint_trip.begin(), Constraint_trip.end());
186
187 // Generate sparse matrix for general (equality) constraints nodal energy conservation and source / sink summation
188 Eigen::VectorXpd non_zero_num(constrant_num);
189 non_zero_num << Connection_num + Eigen::VectorXpd::Ones(International_Market.network.num_vertice), Eigen::VectorXpd::Constant(International_Market.network.num_vertice, International_Market.price_intervals + 3);
190 alglib::integer_1d_array row_sizes_general;
191 row_sizes_general.setcontent(non_zero_num.size(), non_zero_num.data());
192 alglib::sparsematrix constraint_general;
193 alglib::sparsecreatecrs(constrant_num, variable_num, row_sizes_general, constraint_general);
194 for(int row_iter = 0; row_iter < Constraint.outerSize(); ++ row_iter){
195 for(Eigen::SparseMatrix<double, Eigen::RowMajor>::InnerIterator inner_iter(Constraint, row_iter); inner_iter; ++ inner_iter){
196 alglib::sparseset(constraint_general, inner_iter.row(), inner_iter.col(), inner_iter.value());
197 }
198 }
199
200 // -------------------------------------------------------------------------------
201 // Set bounds for general and box constraints
202 // -------------------------------------------------------------------------------
203 Eigen::MatrixXd bound_general = Eigen::MatrixXd::Zero(constrant_num, 2);
204
205 // Bounds of general constraints
206 alglib::real_1d_array lb_general;
207 alglib::real_1d_array ub_general;
208 lb_general.setcontent(bound_general.rows(), bound_general.col(0).data());
209 ub_general.setcontent(bound_general.rows(), bound_general.col(1).data());
210
211 // -------------------------------------------------------------------------------
212 // Set scale of variables
213 // -------------------------------------------------------------------------------
214 Eigen::VectorXd scale_vec = Eigen::VectorXd::Ones(variable_num);
215 alglib::real_1d_array scale;
216 scale.setcontent(scale_vec.size(), scale_vec.data());
217
218 // -------------------------------------------------------------------------------
219 // Set objective coefficients of variables
220 // -------------------------------------------------------------------------------
221 Eigen::VectorXd obj_vec = Eigen::VectorXd::Zero(variable_num);
222 for(int node_iter = 0; node_iter < International_Market.network.num_vertice; ++ node_iter){
223 int row_start = International_Market.network.num_edges + International_Market.network.num_vertice + node_iter * (International_Market.price_intervals + 2);
224 obj_vec.segment(row_start, International_Market.price_intervals + 2) = International_Market.bidded_price_map.bidded_price;
225 }
226 alglib::real_1d_array obj_coeff;
227 obj_coeff.setcontent(obj_vec.size(), obj_vec.data());
228
229 // -------------------------------------------------------------------------------
230 // Set the LP problem object
231 // -------------------------------------------------------------------------------
232 alglib::minlpcreate(variable_num, International_Market.Problem);
233 alglib::minlpsetcost(International_Market.Problem, obj_coeff);
234 alglib::minlpsetlc2(International_Market.Problem, constraint_general, lb_general, ub_general, constrant_num);
235 alglib::minlpsetscale(International_Market.Problem, scale);
236 alglib::minlpsetalgodss(International_Market.Problem, 0.);
237}

302void power_market::International_Market_Optimization(int tick, market_inform &Market){
303 // Update of submitted supply and demand bids at other nations
304 // Bidding zones of other nations: assume inflexible supply or demand
305 Market.submitted_supply.rightCols(Market.num_zone - Market.cross_border_zone_start) = Market.merit_order_curve.rightCols(Market.num_zone - Market.cross_border_zone_start);
306 for(int zone_ID = Market.cross_border_zone_start; zone_ID < Market.num_zone; ++ zone_ID){
307 if(Market.demand_default(tick, zone_ID) >= 0){
308 Market.submitted_demand(Market.price_intervals, zone_ID) += Market.demand_default(tick, zone_ID);
309 }
310 else{
311 Market.submitted_supply(0, zone_ID) += -Market.demand_default(tick, zone_ID);
312 }
313 }
314
315 // -------------------------------------------------------------------------------
316 // Update bounds for box constraints
317 // -------------------------------------------------------------------------------
318 int variable_num = Market.network.num_edges + Market.network.num_vertice * (Market.price_intervals + 3);
319 Eigen::MatrixXd bound_box = Eigen::MatrixXd::Zero(variable_num, 2);
320 bound_box.col(0).head(Market.network.num_edges) = -Market.network.power_constraint.col(1);
321 bound_box.col(1).head(Market.network.num_edges) = Market.network.power_constraint.col(0);
322 bound_box.col(0).segment(Market.network.num_edges, Market.network.num_vertice) = Eigen::VectorXd::Constant(Market.network.num_vertice, -std::numeric_limits<double>::infinity());
323 bound_box.col(1).segment(Market.network.num_edges, Market.network.num_vertice) = Eigen::VectorXd::Constant(Market.network.num_vertice, std::numeric_limits<double>::infinity());
324 for(int node_iter = 0; node_iter < Market.network.num_vertice; ++ node_iter){
325 int row_start = Market.network.num_edges + Market.network.num_vertice + node_iter * (Market.price_intervals + 2);
326 bound_box.middleRows(row_start, Market.price_intervals + 2).col(0) = -Market.submitted_demand.col(node_iter);
327 bound_box.middleRows(row_start, Market.price_intervals + 2).col(1) = Market.submitted_supply.col(node_iter);
328 }
329
330 // Bounds of box constraints
331 alglib::real_1d_array lb_box;
332 alglib::real_1d_array ub_box;
333 lb_box.setcontent(bound_box.rows(), bound_box.col(0).data());
334 ub_box.setcontent(bound_box.rows(), bound_box.col(1).data());
335 alglib::minlpsetbc(Market.Problem, lb_box, ub_box);
336
337 // -------------------------------------------------------------------------------
338 // Solve the problem and get results
339 // -------------------------------------------------------------------------------
340 alglib::minlpoptimize(Market.Problem);
341 alglib::real_1d_array sol;
342 alglib::minlpreport rep;
343 alglib::minlpresults(Market.Problem, sol, rep);
344 Eigen::VectorXd sol_vec = Eigen::Map <Eigen::VectorXd> (sol.getcontent(), sol.length());
345
346 for(int node_iter = 0; node_iter < Market.network.num_vertice; ++ node_iter){
347 // Store power source / sink
348 int row_start = Market.network.num_edges + Market.network.num_vertice + node_iter * (Market.price_intervals + 2);
349 Market.confirmed.supply(tick, node_iter) = (sol_vec.segment(row_start, Market.price_intervals + 2).array().max(0)).sum();
350 Market.confirmed.demand(tick, node_iter) = -(sol_vec.segment(row_start, Market.price_intervals + 2).array().min(0)).sum();
351
352 // Store nodal prices
353 Market.confirmed.price(tick, node_iter) = Market.bidded_price_map.bidded_price(0) + rep.lagbc[row_start];
354 Market.confirmed.price(tick, node_iter) = int(Market.confirmed.price(tick, node_iter)) + .5;
355 if(Market.confirmed.price(tick, node_iter) < Market.bidded_price_map.bidded_price(1)){

357 }
358 else if(Market.confirmed.price(tick, node_iter) > Market.bidded_price_map.bidded_price(Market.price_intervals)){
359 Market.confirmed.price(tick, node_iter) = Market.bidded_price_map.bidded_price(Market.price_intervals + 1);
360 }
361
362 // Store ratio at nodes
363 for(int price_iter = 0; price_iter < Market.price_intervals + 2; ++ price_iter){
364 if(Market.bidded_price_map.bidded_price(price_iter) >= Market.confirmed.price(tick, node_iter) || price_iter == Market.price_intervals + 1){
365 if(sol[row_start + price_iter] >= 0.){
366 Market.confirmed.ratio_demand(node_iter) = std::min(Market.submitted_demand(price_iter, node_iter), Market.submitted_supply(price_iter, node_iter) - sol[row_start + price_iter]);
367 Market.confirmed.ratio_supply(node_iter) = Market.confirmed.ratio_demand(node_iter) + sol[row_start + price_iter];
368 Market.confirmed.ratio_demand(node_iter) /= Market.submitted_demand(price_iter, node_iter) + 1E-12;
369 Market.confirmed.ratio_supply(node_iter) /= Market.submitted_supply(price_iter, node_iter) + 1E-12;
370 }
371 else{
372 Market.confirmed.ratio_supply(node_iter) = std::min(Market.submitted_supply(price_iter, node_iter), Market.submitted_demand(price_iter, node_iter) + sol[row_start + price_iter]);
373 Market.confirmed.ratio_demand(node_iter) = Market.confirmed.ratio_supply(node_iter) - sol[row_start + price_iter];
374 Market.confirmed.ratio_demand(node_iter) /= Market.submitted_demand(price_iter, node_iter) + 1E-12;
375 Market.confirmed.ratio_supply(node_iter) /= Market.submitted_supply(price_iter, node_iter) + 1E-12;
376 }
377 break;
378 }
379 }
380 }
381
382 // Store cross-border transmission flow
383 Market.network.confirmed_power.row(tick) = sol_vec.head(Market.network.num_edges);
