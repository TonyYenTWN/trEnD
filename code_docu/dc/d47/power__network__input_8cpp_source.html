8 void cbt_data_input(power_network::network_inform &Power_network_inform, std::string fin_cbt, std::string fin_entry){
9 // Read cross-border transmission data
10 auto fin_cbt_dim = basic::get_file_dim(fin_cbt);
11 auto fin_entry_dim = basic::get_file_dim(fin_entry);
12
13 // Store bidding zone names
14 Power_network_inform.cbt.bz_names = basic::get_col_name(fin_cbt, fin_cbt_dim[1]);
15
16 // Store entry points and corresponding bidding zones
17 // entry_bz = ID for the corresponding external bidding zone
18 Power_network_inform.cbt.entry_nodes = basic::read_file(fin_entry_dim[0], fin_entry_dim[1], fin_entry);
19 Power_network_inform.cbt.entry_bz = Eigen::VectorXi(fin_entry_dim[0]);
20 for(int edge_iter = 0; edge_iter < fin_entry_dim[0]; ++ edge_iter){
21 Power_network_inform.cbt.entry_bz(edge_iter) = int(Power_network_inform.cbt.entry_nodes(edge_iter, 0));
22 }
23 Power_network_inform.cbt.entry_bz = Power_network_inform.cbt.entry_bz.array().max(0) - 1;
24 Power_network_inform.cbt.entry_nodes = Power_network_inform.cbt.entry_nodes.rightCols(fin_entry_dim[1] - 1);
25 Power_network_inform.cbt.entry_nodes = Power_network_inform.cbt.entry_nodes.array().max(0) - 1.;
26
27 // Find number of entry nodes for each bidding zones
28 Power_network_inform.cbt.entry_node_num = Eigen::VectorXi::Zero(fin_entry_dim[0]);
29 for(int zone_iter = 0; zone_iter < fin_entry_dim[0]; ++ zone_iter){
30 for(int node_iter = 0; node_iter < fin_entry_dim[1] - 1; ++ node_iter){
31 if(Power_network_inform.cbt.entry_nodes(zone_iter, node_iter) < 0.){
32 break;
33 }
34 Power_network_inform.cbt.entry_node_num(zone_iter) += 1;
35 }
36 }
37
38 // Store cbt constraints
39 Power_network_inform.cbt.flow_constraint = basic::read_file(fin_cbt_dim[0], fin_cbt_dim[1], fin_cbt);
40 }

191// Find the distance and covariance between points
192void power_network::point_distance_cov(points_struct &point, double lambda){
193 double pi = boost::math::constants::pi<double>();
194 double tol = 1E-8;
195 Eigen::MatrixXd covariance = Eigen::MatrixXd::Ones(point.lon.size(), point.lon.size());
196 point.distance = Eigen::MatrixXd::Zero(point.lon.size(), point.lon.size());
197
198 #pragma omp parallel
199 {
200 #pragma omp for
201 for(int row_iter = 0; row_iter < point.lon.size() - 1; ++ row_iter){
202 for(int col_iter = row_iter + 1; col_iter < point.lon.size(); ++ col_iter){
203 point.distance(row_iter, col_iter) = spatial_field::geodist(Eigen::Vector2d(point.lon(row_iter) * pi / 180., point.lat(row_iter) * pi / 180.), Eigen::Vector2d(point.lon(col_iter) * pi / 180., point.lat(col_iter) * pi / 180.));
204 point.distance(col_iter, row_iter) = point.distance(row_iter, col_iter);
205 covariance(row_iter, col_iter) = exp(-pow(point.distance(row_iter, col_iter), 1.) / point.grid_length / lambda);
206 covariance(col_iter, row_iter) = covariance(row_iter, col_iter);
207 }
208 }
209 }
210
211 point.covariance = covariance.sparseView(tol);
212}
