104 stored_information[stored_information.size() - 1](0) = Power_market_inform.agent_profiles.cross_border[edge_iter].profiles[node_iter].node_ID;
105 }
106 }
107
108 Eigen::MatrixXd Output_data = Eigen::MatrixXd::Zero(stored_information.size(), var_num);
109 for(int row_iter = 0; row_iter < stored_information.size(); ++ row_iter){
110 Output_data.row(row_iter) = stored_information[row_iter];
