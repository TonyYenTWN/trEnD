9agent::sorted_vector agent::sort(Eigen::VectorXd original){
10 // Sort of vector
11 std::vector<int> item_seq(original.size());
12 std::iota(item_seq.begin(), item_seq.end(), 0);
13 std::sort(item_seq.begin(), item_seq.end(), [&](int i,int j){return original(i) < original(j);});
14
15 // Output of vector
16 agent::sorted_vector result;
17 result.id = Eigen::VectorXi(original.size());
18 result.value = Eigen::VectorXd(original.size());
19 for(int item_ID = 0; item_ID < original.size(); ++ item_ID){
20 result.id(item_ID) = item_seq[item_ID];
21 result.value(item_ID) = original(item_seq[item_ID]);
22 }
23
24 return(result);
25}

27void agent::end_user::smart_appliance_schedule(agent::sorted_vector sorted_tariff, Eigen::VectorXd normalized_default_demand_profile, agent::end_user::smart_appliance_inform &result){
28 // Initialization
29 double total_flex_demand_energy = result.scale * normalized_default_demand_profile.sum() + result.unfulfilled_demand;
30 double flex_demand_capacity_max = total_flex_demand_energy / normalized_default_demand_profile.size() / result.flexibility_factor;
31 int flex_demand_duration = int(result.flexibility_factor * normalized_default_demand_profile.size());
32 if(double(result.flexibility_factor * normalized_default_demand_profile.size() - flex_demand_duration) != 0){
33 flex_demand_duration += 1;
34 }
35
36 // Schedule the demand to low price periods
37 result.normalized_scheduled_profile = Eigen::VectorXd::Zero(normalized_default_demand_profile.size());
38 for(int tick = 0; tick < flex_demand_duration - 1; ++ tick){
39 result.normalized_scheduled_profile(sorted_tariff.id(tick)) = flex_demand_capacity_max;
40 }
41 result.normalized_scheduled_profile(sorted_tariff.id(flex_demand_duration - 1)) = total_flex_demand_energy - (flex_demand_duration - 1) * flex_demand_capacity_max;
42}
