74sorted_vector sort(Eigen::VectorXd original){
75 // Sort of vector
76 std::vector<int> item_seq(original.size());
77 std::iota(item_seq.begin(), item_seq.end(), 0);
78 sort(item_seq.begin(), item_seq.end(), [&](int i,int j){return original(i) < original(j);});
79
80 // Output of vector
81 sorted_vector result;
82 result.id = Eigen::VectorXi(original.size());
83 result.value = Eigen::VectorXd(original.size());
84 for(int item_ID = 0; item_ID < original.size(); ++ item_ID){
85 result.id(item_ID) = item_seq[item_ID];
86 result.value(item_ID) = original(item_seq[item_ID]);
87 }
88
89 return(result);
90}

92void smart_appliance_schedule(sorted_vector sorted_tariff, Eigen::VectorXd normalized_default_demand_profile, double residual_unfulfilled_demand, smart_appliance_inform &result){
93 // Initialization
94 double total_flex_demand_energy = result.scale * normalized_default_demand_profile.sum() + residual_unfulfilled_demand;
95 double flex_demand_capacity_max = total_flex_demand_energy / normalized_default_demand_profile.size() / result.flexibility_factor;
96 int flex_demand_duration = int(result.flexibility_factor * normalized_default_demand_profile.size());
97 if(double(result.flexibility_factor * normalized_default_demand_profile.size() - flex_demand_duration) != 0){
98 flex_demand_duration += 1;
99 }
100
101 // Schedule the demand to low price periods
102 result.normalized_scheduled_profile = Eigen::VectorXd::Zero(normalized_default_demand_profile.size());
103 for(int tick = 0; tick < flex_demand_duration - 1; ++ tick){
104 result.normalized_scheduled_profile(sorted_tariff.id(tick)) = flex_demand_capacity_max;
105 }
106 result.normalized_scheduled_profile(sorted_tariff.id(flex_demand_duration - 1)) = total_flex_demand_energy - (flex_demand_duration - 1) * flex_demand_capacity_max;
107}

213void storage_schedule_naive(sorted_vector sorted_tariff, storage_inform &result){
214 // Initialization
215 int total_duration = sorted_tariff.value.size();
216 int ch_surplus_duration = int(std::max(result.soc_final - result.soc_ini, 0.) / result.capacity_scale);
217 int dc_surplus_duration = int(std::max(result.soc_ini - result.soc_final, 0.) / result.capacity_scale);
218 result.normalized_scheduled_capacity_profile = Eigen::VectorXd::Zero(total_duration);
219 result.normalized_scheduled_soc_profile = result.soc_ini * Eigen::VectorXd::Ones(total_duration);
220 // Allowed exchange matrix: row = charge period; col = discharge period
221 Eigen::MatrixXi Allowed_Exchange = Eigen::MatrixXi::Ones(total_duration, total_duration);
222
223 // Schedule of surplus dc / ch so that the soc change is close to as expected
224 if(ch_surplus_duration > 0){
225 for(int tick = 0; tick < ch_surplus_duration; ++ tick){
226 result.normalized_scheduled_capacity_profile(sorted_tariff.id(tick)) = -result.capacity_scale / result.efficiency;
227 for(int tock = sorted_tariff.id(tick); tock < total_duration; ++ tock){
228 result.normalized_scheduled_soc_profile(tock) += result.capacity_scale;
229 }
230 Allowed_Exchange.row(sorted_tariff.id(tick)) = Eigen::VectorXi::Zero(total_duration);
231 Allowed_Exchange.col(sorted_tariff.id(tick)) = Eigen::VectorXi::Zero(total_duration);
232 }
233 }
234 else{
235 for(int tick = 0; tick < dc_surplus_duration; ++ tick){
236 result.normalized_scheduled_capacity_profile(sorted_tariff.id(total_duration - tick - 1)) = result.capacity_scale * result.efficiency;
237 for(int tock = sorted_tariff.id(total_duration - tick - 1); tock < total_duration; ++ tock){
238 result.normalized_scheduled_soc_profile(tock) -= result.capacity_scale;
239 }
240 Allowed_Exchange.row(sorted_tariff.id(total_duration - tick - 1)) = Eigen::VectorXi::Zero(total_duration);
241 Allowed_Exchange.col(sorted_tariff.id(total_duration - tick - 1)) = Eigen::VectorXi::Zero(total_duration);
242 }
243 }
244
245 // Check profitability of exchange
246 for(int tick = 0; tick < total_duration; ++ tick){
247 for(int tock = 0; tock < total_duration; ++ tock){
248 if(Allowed_Exchange(sorted_tariff.id(tick), sorted_tariff.id(tock)) == 1){
249 if(sorted_tariff.value(tick) >= pow(result.efficiency, 2) * sorted_tariff.value(tock)){
250 Allowed_Exchange(sorted_tariff.id(tick), sorted_tariff.id(tock)) = 0;
251 }
252 }
253 }
254 }
255
256 // Pair 2 time intervals for maximum possible price arbitrage
257 double maximum_price_diff;
258 Eigen::Vector2i maximum_price_diff_ID;
259 Eigen::MatrixXi Allowed_Exchange_temp = Allowed_Exchange;
260 Eigen::MatrixXd soc_profile_temp;
261 int count = 0;
262 while(Allowed_Exchange_temp.sum() > 0){
263 //while(count < 10){
264 count += 1;
265 maximum_price_diff = 0;
266 Allowed_Exchange_temp = Allowed_Exchange;
267
268 // Check if the excahnge pair is feasible and if yes, most profitable up to point
269 for(int tick = 0; tick < total_duration; ++ tick){
270 for(int tock = 0; tock < total_duration; ++ tock){
271 if(Allowed_Exchange(sorted_tariff.id(tick), sorted_tariff.id(tock)) == 1){
272 soc_profile_temp = result.normalized_scheduled_soc_profile;
273 if(sorted_tariff.id(tick) < sorted_tariff.id(tock)){
274 for(int tuck = sorted_tariff.id(tick); tuck < sorted_tariff.id(tock); ++ tuck){
275 soc_profile_temp(tuck) += result.capacity_scale;
276 if(soc_profile_temp(tuck) > result.energy_scale){
277 Allowed_Exchange_temp(sorted_tariff.id(tick), sorted_tariff.id(tock)) = 0;
278 break;
279 }
280 }
281 }
282 else{
283 for(int tuck = sorted_tariff.id(tock); tuck < sorted_tariff.id(tick); ++ tuck){
284 soc_profile_temp(tuck) -= result.capacity_scale;
285 if(soc_profile_temp(tuck) < 0){
286 Allowed_Exchange_temp(sorted_tariff.id(tick), sorted_tariff.id(tock)) = 0;
287 break;
288 }
289 }
290 }
291
292 if(Allowed_Exchange_temp(sorted_tariff.id(tick), sorted_tariff.id(tock)) == 1){
293 if(sorted_tariff.value(tock) * result.efficiency - sorted_tariff.value(tick) / result.efficiency > maximum_price_diff){
294 maximum_price_diff = sorted_tariff.value(tock) * result.efficiency - sorted_tariff.value(tick) / result.efficiency;
295 maximum_price_diff_ID << sorted_tariff.id(tick), sorted_tariff.id(tock);
296 }
297 }
298 }
299 }
300 }
301
302 // Update if feasible price arbitrage between pairs exists
303 if(Allowed_Exchange_temp.sum() > 0){
304 result.normalized_scheduled_capacity_profile(maximum_price_diff_ID(0)) = -result.capacity_scale / result.efficiency;
305 result.normalized_scheduled_capacity_profile(maximum_price_diff_ID(1)) = result.capacity_scale * result.efficiency;
306
307 if(maximum_price_diff_ID(0) < maximum_price_diff_ID(1)){
308 for(int tuck = maximum_price_diff_ID(0); tuck < maximum_price_diff_ID(1); ++ tuck){
309 result.normalized_scheduled_soc_profile(tuck) += result.capacity_scale;
310 }
311 }
312 else{
313 for(int tuck = maximum_price_diff_ID(1); tuck < maximum_price_diff_ID(0); ++ tuck){
314 result.normalized_scheduled_soc_profile(tuck) -= result.capacity_scale;
315 }
316 }
317
318 Allowed_Exchange.row(maximum_price_diff_ID(0)) = Eigen::VectorXi::Zero(total_duration);
319 Allowed_Exchange.col(maximum_price_diff_ID(0)) = Eigen::VectorXi::Zero(total_duration);
320 Allowed_Exchange.row(maximum_price_diff_ID(1)) = Eigen::VectorXi::Zero(total_duration);
321 Allowed_Exchange.col(maximum_price_diff_ID(1)) = Eigen::VectorXi::Zero(total_duration);
322 }
323 }
