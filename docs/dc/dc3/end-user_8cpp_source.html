6 // -------------------------------------------------------------------------------
7 // LP Solver initialization for operation strategy of end-users
8 // Warm-up once and reuse for the rest of time slices
9 // Variables are sorted as the following order (repeat for every t)
10 // U^s(t), U^d(t): aggregated scheduled supply and demand
11 // U^b(t): electricity demand from BESS
12 // U^ev(t): electricity demand from EV
13 // U^sa(t): electricity demand from smart appliance
14 // RL_0(t): default residual load
15 // ch^b(t): electricity charge from BESS
16 // dc^b(t): electricity discharge from BESS
17 // d^b(t): self consumption from BESS
18 // soc^b(t): state of charge of BESS
19 // ch^ev(t): electricity charge from EV
20 // dc^ev(t): electricity discharge from EV
21 // d^ev(t): electricity demand from EV
22 // soc^ev(t): state of charge of EV
23 // d^sa(..., t): scheduled smart appliance load at time t
24 // d^sa(t): default smart appliance load at time t
25 // -------------------------------------------------------------------------------
26 int foresight_time = profile.operation.foresight_time;
27 int load_shift_time = profile.operation.smart_appliance.shift_time;
28
29 // -------------------------------------------------------------------------------
30 // Set matrix for general constraints
31 // -------------------------------------------------------------------------------
32 // Generate sparse matrix for general equality constraints of dynamic equation
33 int constrant_num = 7 * foresight_time + load_shift_time;
34 int variable_per_time = 14 + foresight_time + load_shift_time;
35 int variable_num = variable_per_time * foresight_time + foresight_time + load_shift_time;
36 Eigen::VectorXpd non_zero_num(constrant_num);
37 non_zero_num.head(5 * foresight_time) << Eigen::VectorXpd::Constant(foresight_time, 6), Eigen::VectorXpd::Constant(foresight_time, 3), 4, Eigen::VectorXpd::Constant(foresight_time - 1, 5), Eigen::VectorXpd::Constant(foresight_time, 3), 4, Eigen::VectorXpd::Constant(foresight_time - 1, 5);
38 non_zero_num.segment(5 * foresight_time, foresight_time - load_shift_time) = Eigen::VectorXpd::Constant(foresight_time - load_shift_time, 2 * load_shift_time + 1);
39 if(load_shift_time > 0){
40 for(int tick = 0; tick < load_shift_time; ++ tick){
41 int row_ID = 6 * foresight_time - load_shift_time + tick;
42 non_zero_num(row_ID) = 2 * load_shift_time - tick;
43 }
44
45 for(int tick = 0; tick < foresight_time + load_shift_time; ++ tick){
46 int row_ID = 6 * foresight_time + tick;
47 int tock_start = std::max(tick - 2 * load_shift_time, 0);
48 int tock_end = std::min(tick, foresight_time - 1);
49 non_zero_num(row_ID) = tock_end - tock_start + 1;
50 }
51 }
52 non_zero_num.segment(5 * foresight_time, foresight_time) += Eigen::VectorXpd::Ones(foresight_time);
53 non_zero_num.tail(foresight_time + load_shift_time) += Eigen::VectorXpd::Ones(foresight_time + load_shift_time);
54 alglib::integer_1d_array row_sizes_general;
55 row_sizes_general.setcontent(non_zero_num.size(), non_zero_num.data());
56 alglib::sparsematrix constraint_general;
57 alglib::sparsecreatecrs(constrant_num, variable_num, row_sizes_general, constraint_general);

59 // Fill in the coefficients for the sparse matrix
60 // U^d(t) - U^s(t) - U^b(t) - U^ev(t) - U^sa(t) - RL_0(t) = 0
61 for(int row_iter = 0; row_iter < foresight_time; ++ row_iter){
62 int U_d_ID = row_iter * variable_per_time;
63 int U_s_ID = row_iter * variable_per_time + 1;
64 int U_b_ID = row_iter * variable_per_time + 2;
65 int U_ev_ID = row_iter * variable_per_time + 3;
66 int U_sa_ID = row_iter * variable_per_time + 4;
67 int RL_ID = row_iter * variable_per_time + 5;
68

70 alglib::sparseset(constraint_general, row_iter, U_s_ID, -1.);
71 alglib::sparseset(constraint_general, row_iter, U_b_ID, -1.);
72 alglib::sparseset(constraint_general, row_iter, U_ev_ID, -1.);
73 alglib::sparseset(constraint_general, row_iter, U_sa_ID, -1.);
74 alglib::sparseset(constraint_general, row_iter, RL_ID, -1.);
75 }

77 // U^b(t) - 1 / eta * ch^b(t) + eta * dc^b(t) = 0
78 for(int row_iter = 0; row_iter < foresight_time; ++ row_iter){
79 int row_ID = foresight_time + row_iter;
80 int U_b_ID = row_iter * variable_per_time + 2;
81 int ch_b_ID = row_iter * variable_per_time + 6;
82 int dc_b_ID = row_iter * variable_per_time + 7;

85 alglib::sparseset(constraint_general, row_ID, ch_b_ID, -1. / profile.operation.BESS.efficiency);
86 alglib::sparseset(constraint_general, row_ID, dc_b_ID, profile.operation.BESS.efficiency);
87 }

89 // soc^b(t) - soc^b(t - 1) - ch^b(t) + dc^b(t) + d^b(t) = 0

104 int row_ID = 2 * foresight_time + row_iter;
105 int s_b_prev_ID = (row_iter - 1) * variable_per_time + 9;
106 int ch_b_ID = row_iter * variable_per_time + 6;
107 int dc_b_ID = row_iter * variable_per_time + 7;

144 }
145 for(int row_iter = 1; row_iter < foresight_time; ++ row_iter){
146 int row_ID = 4 * foresight_time + row_iter;

172 }
173 }

176 for(int row_iter = 0; row_iter < foresight_time + load_shift_time; ++ row_iter){
177 int row_ID = 6 * foresight_time + row_iter;
178
179 for(int tick = 0; tick < 2 * load_shift_time + 1; ++ tick){
180 int tick_ID = row_iter - 2 * load_shift_time + tick;
181 if(tick_ID >= 0 && tick_ID < foresight_time){
182 int d_sa_ID = tick_ID * variable_per_time + 14 + row_iter;
183 alglib::sparseset(constraint_general, row_ID, d_sa_ID, -1.);
184 }
185 }
186
187 int d_sa_total_ID = variable_per_time * foresight_time + row_iter;
188 alglib::sparseset(constraint_general, row_ID, d_sa_total_ID, 1.);
189 }

191 // -------------------------------------------------------------------------------
192 // Set bounds for general constraints
193 // -------------------------------------------------------------------------------
194 Eigen::MatrixXd bound_general = Eigen::MatrixXd::Zero(constrant_num, 2);
195 alglib::real_1d_array lb_general;
196 alglib::real_1d_array ub_general;
197 lb_general.setcontent(bound_general.rows(), bound_general.col(0).data());
198 ub_general.setcontent(bound_general.rows(), bound_general.col(1).data());

200 // -------------------------------------------------------------------------------

202 // -------------------------------------------------------------------------------
203 alglib::minlpcreate(variable_num, profile.operation.Problem);
204 alglib::minlpsetlc2(profile.operation.Problem, constraint_general, lb_general, ub_general, constrant_num);

208void agent::end_user::end_user_LP_optimize(int tick, profile &profile){
209 int foresight_time = profile.operation.foresight_time;
210 int load_shift_time = profile.operation.smart_appliance.shift_time;
211 int price_interval = power_market::parameters::price_interval();
212 int variable_per_time = 14 + foresight_time + load_shift_time;
213 int variable_num = variable_per_time * foresight_time + foresight_time + load_shift_time;
214
215 // -------------------------------------------------------------------------------
216 // Set bounds for box constraints
217 // -------------------------------------------------------------------------------
218 Eigen::MatrixXd bound_box = Eigen::MatrixXd::Zero(variable_num, 2);
219 for(int tock = 0; tock < foresight_time; ++ tock){
220 int row_start = tock * variable_per_time;
221 int U_d_ID = row_start;
222 int U_s_ID = row_start + 1;
223 int U_b_ID = row_start + 2;
224 int U_ev_ID = row_start + 3;
225 int U_sa_ID = row_start + 4;
226 int RL_ID = row_start + 5;
227 int ch_b_ID = row_start + 6;
228 int dc_b_ID = row_start + 7;
229 int d_b_ID = row_start + 8;
230 int s_b_ID = row_start + 9;
231 int ch_ev_ID = row_start + 10;
232 int dc_ev_ID = row_start + 11;
233 int d_ev_ID = row_start + 12;
234 int s_ev_ID = row_start + 13;
235
236 bound_box.row(U_d_ID) << -std::numeric_limits<double>::infinity(), std::numeric_limits<double>::infinity();
237 bound_box.row(U_s_ID) << -std::numeric_limits<double>::infinity(), std::numeric_limits<double>::infinity();
238 bound_box.row(U_b_ID) << -std::numeric_limits<double>::infinity(), std::numeric_limits<double>::infinity();
239 bound_box.row(U_ev_ID) << -std::numeric_limits<double>::infinity(), std::numeric_limits<double>::infinity();
240 bound_box.row(U_sa_ID) << -std::numeric_limits<double>::infinity(), std::numeric_limits<double>::infinity();
241 bound_box.row(RL_ID) << Eigen::RowVector2d::Constant(profile.operation.default_demand_profile(tock) - profile.operation.default_PV_profile(tock));
242 bound_box.row(ch_b_ID) << 0., profile.operation.BESS.capacity_scale;
243 bound_box.row(ch_b_ID) *= profile.investment.decision.BESS;
244 bound_box.row(dc_b_ID) << 0., profile.operation.BESS.capacity_scale;
245 bound_box.row(dc_b_ID) *= profile.investment.decision.BESS;
246 bound_box.row(s_b_ID) << 0., profile.operation.BESS.energy_scale;
247 bound_box.row(d_b_ID) << Eigen::RowVector2d::Constant(profile.operation.BESS.self_consumption);
248 bound_box.row(ch_ev_ID) << 0., profile.operation.EV.BESS.capacity_scale;
249 bound_box.row(ch_ev_ID) *= profile.investment.decision.EV_self_charging;
250 bound_box.row(dc_ev_ID) << 0., profile.operation.EV.BESS.capacity_scale;
251 bound_box.row(dc_ev_ID) *= profile.investment.decision.EV_self_charging;
252 bound_box.row(s_ev_ID) << 0., profile.operation.EV.BESS.energy_scale;
253 bound_box.row(d_ev_ID) << Eigen::RowVector2d::Constant(profile.operation.EV.default_demand_profile(tock) + profile.operation.EV.BESS.self_consumption);
254 if(tock == 0){
255 bound_box.row(d_b_ID) -= Eigen::RowVector2d::Constant(profile.operation.BESS.soc);
256 bound_box.row(d_ev_ID) -= Eigen::RowVector2d::Constant(profile.operation.EV.BESS.soc);
257 }
258
259 for(int tuck = 0; tuck < 2 * load_shift_time + 1; ++ tuck){
260 int tuck_ID = tock - load_shift_time + tuck;
261 if(tuck_ID >= -load_shift_time && tuck_ID < foresight_time){
262 int d_sa_ID = tock * variable_per_time + 14 + tuck_ID + load_shift_time;
263 bound_box.row(d_sa_ID) << 0., profile.operation.smart_appliance.unfulfilled_demand(tuck_ID + load_shift_time);
264 //std::cout << bound_box.row(d_sa_ID) << "\n";
265 }
266 }
267 //std::cout << "\n";
268 }
269
270 for(int tock = 0; tock < foresight_time + load_shift_time; ++ tock){
271 int d_sa_total_ID = variable_per_time * foresight_time + tock;
272 bound_box.row(d_sa_total_ID) = Eigen::RowVector2d::Constant(profile.operation.smart_appliance.unfulfilled_demand(tock));
273
274 }
275
276 // Bounds of box constraints
277 alglib::real_1d_array lb_box;
278 alglib::real_1d_array ub_box;
279 lb_box.setcontent(bound_box.rows(), bound_box.col(0).data());
280 ub_box.setcontent(bound_box.rows(), bound_box.col(1).data());
281 alglib::minlpsetbc(profile.operation.Problem, lb_box, ub_box);
282
283 // -------------------------------------------------------------------------------
284 // Set objective coefficients of variables
285 // -------------------------------------------------------------------------------
286 Eigen::VectorXd obj_vec = Eigen::VectorXd::Zero(variable_num);
287 for(int tock = 0; tock < foresight_time; ++ tock){
288 int row_start = tock * variable_per_time;
289 int U_d_ID = row_start;
290 int U_s_ID = row_start + 1;
291
292 obj_vec(U_d_ID) = profile.operation.price_demand_profile(tock);
293 obj_vec(U_s_ID) = -profile.operation.price_supply_profile(tock);
294 }
295 alglib::real_1d_array obj_coeff;
296 obj_coeff.setcontent(obj_vec.size(), obj_vec.data());
297 alglib::minlpsetcost(profile.operation.Problem, obj_coeff);
298
299 // -------------------------------------------------------------------------------
300 // Solve the problem and store the results
301 // -------------------------------------------------------------------------------
302 alglib::minlpoptimize(profile.operation.Problem);
303 alglib::real_1d_array sol;
304 alglib::minlpreport rep;
305 alglib::minlpresults(profile.operation.Problem, sol, rep);
306 profile.operation.BESS.scheduled_capacity = sol[2];
307 profile.operation.EV.BESS.scheduled_capacity = sol[3];
308 profile.operation.smart_appliance.scheduled_demand = Eigen::VectorXd::Zero(2 * load_shift_time + 1);
309 for(int tock = 0; tock < 2 * load_shift_time + 1; ++ tock){
310 profile.operation.smart_appliance.scheduled_demand(tock) = sol[14 + tock];
311 }
312
313 int price_demand_inflex_ID = price_interval + 1;
314 int price_demand_flex_ID = price_interval + 1;
315 int price_supply_inflex_ID = 0;
316 int price_supply_flex_ID = 0;
317 profile.operation.bids.submitted_demand_inflex(price_demand_inflex_ID) += std::max(sol[5], 0.);
318 profile.operation.bids.submitted_supply_inflex(0) += -std::min(sol[5], 0.);
319 profile.operation.bids.submitted_demand_inflex(price_supply_inflex_ID) += sol[14];
320 if(profile.investment.decision.redispatch){
321 profile.operation.bids.submitted_demand_flex(price_demand_flex_ID) += sol[4] - sol[14];
322 profile.operation.bids.submitted_demand_flex(price_demand_flex_ID) += std::max(sol[2], 0.);
323 profile.operation.bids.submitted_supply_flex(price_supply_flex_ID) += -std::min(sol[2], 0.);

326 }

330 profile.operation.bids.submitted_supply_inflex(price_supply_flex_ID) += -std::min(sol[2], 0.);
331 profile.operation.bids.submitted_demand_inflex(price_demand_flex_ID) += std::max(sol[3], 0.);
332 profile.operation.bids.submitted_supply_inflex(price_supply_flex_ID) += -std::min(sol[3], 0.);
