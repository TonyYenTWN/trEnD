9agent::sorted_vector agent::sort(Eigen::VectorXd original){
10 // Sort of vector
11 std::vector<int> item_seq(original.size());
12 std::iota(item_seq.begin(), item_seq.end(), 0);
13 std::sort(item_seq.begin(), item_seq.end(), [&](int i,int j){return original(i) < original(j);});
14
15 // Output of vector
16 agent::sorted_vector result;
17 result.id = Eigen::VectorXi(original.size());
18 result.value = Eigen::VectorXd(original.size());
19 for(int item_ID = 0; item_ID < original.size(); ++ item_ID){
20 result.id(item_ID) = item_seq[item_ID];
21 result.value(item_ID) = original(item_seq[item_ID]);
22 }
23
24 return(result);
25}

27void agent::end_user::smart_appliance_schedule(agent::sorted_vector sorted_tariff, Eigen::VectorXd normalized_default_demand_profile, agent::end_user::smart_appliance_inform &result){
28 // Initialization
29 double total_flex_demand_energy = result.scale * normalized_default_demand_profile.sum() + result.unfulfilled_demand;
30 double flex_demand_capacity_max = total_flex_demand_energy / normalized_default_demand_profile.size() / result.flexibility_factor;
31 int flex_demand_duration = int(result.flexibility_factor * normalized_default_demand_profile.size());
32 if(double(result.flexibility_factor * normalized_default_demand_profile.size() - flex_demand_duration) != 0){
33 flex_demand_duration += 1;
34 }
35
36 // Schedule the demand to low price periods
37 result.normalized_scheduled_profile = Eigen::VectorXd::Zero(normalized_default_demand_profile.size());
38 for(int tick = 0; tick < flex_demand_duration - 1; ++ tick){
39 result.normalized_scheduled_profile(sorted_tariff.id(tick)) = flex_demand_capacity_max;
40 }
41 result.normalized_scheduled_profile(sorted_tariff.id(flex_demand_duration - 1)) = total_flex_demand_energy - (flex_demand_duration - 1) * flex_demand_capacity_max;
42}

44alglib::minlpstate agent::end_user::storage_schedule_LP_mold(int foresight_time){
45 alglib::minlpstate Problem;
46
47 // -------------------------------------------------------------------------------
48 // LP Solver initialization for BESS schedule
49 // Warm-up once and reuse for the rest of time slices
50 // Variables are sorted as {s_i, q_self(0), q_sell(0), q_dc(0), q_ch(0), s(0), ...}
51 // -------------------------------------------------------------------------------
52
53 // -------------------------------------------------------------------------------
54 // Set matrix for general constraints
55 // -------------------------------------------------------------------------------
56 // Generate sparse matrix for general equality constraints of dynamic equation
57 int constrant_num = 2 * foresight_time;
58 int variable_num = 5 * foresight_time + 1;
59 Eigen::VectorXpd non_zero_num(constrant_num);
60 non_zero_num << Eigen::VectorXpd::Constant(foresight_time, 4), Eigen::VectorXpd::Constant(foresight_time, 3);
61 alglib::integer_1d_array row_sizes_general;
62 row_sizes_general.setcontent(non_zero_num.size(), non_zero_num.data());
63 alglib::sparsematrix constraint_general;
64 alglib::sparsecreatecrs(constrant_num, variable_num, row_sizes_general, constraint_general);
65
66 // Fill in the coefficients for the sparse matrix
67 // -s(t - 1) + q_dc(t) - q_ch(t) + s(t) = 0
68 for(int row_iter = 0; row_iter < foresight_time; ++ row_iter){
69 int s_prev_ID = 5 * row_iter;
70 int q_dc_ID = 5 * row_iter + 3;
71 int q_ch_ID = 5 * row_iter + 4;
72 int s_ID = 5 * row_iter + 5;
73
74 alglib::sparseset(constraint_general, row_iter, s_prev_ID, -1.);
75 alglib::sparseset(constraint_general, row_iter, q_dc_ID, 1.);
76 alglib::sparseset(constraint_general, row_iter, q_ch_ID, -1.);
77 alglib::sparseset(constraint_general, row_iter, s_ID, 1.);
78 }
79 // q_self(t) + q_sell(t) - q_dc(t) = 0
80 for(int row_iter = 0; row_iter < foresight_time; ++ row_iter){
81 int row_ID = foresight_time + row_iter;
82 int q_self_ID = 5 * row_iter + 1;
83 int q_sell_ID = 5 * row_iter + 2;
84 int q_dc_ID = 5 * row_iter + 3;
85
86 alglib::sparseset(constraint_general, row_ID, q_self_ID, 1.);
87 alglib::sparseset(constraint_general, row_ID, q_sell_ID, 1.);
88 alglib::sparseset(constraint_general, row_ID, q_dc_ID, -1.);
89 }
90
91 // -------------------------------------------------------------------------------
92 // Set bounds for general constraints
93 // -------------------------------------------------------------------------------
94 Eigen::MatrixXd bound_general = Eigen::MatrixXd::Zero(constrant_num, 2);
95 alglib::real_1d_array lb_general;
96 alglib::real_1d_array ub_general;
97 lb_general.setcontent(bound_general.rows(), bound_general.col(0).data());
98 ub_general.setcontent(bound_general.rows(), bound_general.col(1).data());
99
100 // -------------------------------------------------------------------------------
101 // Set the LP problem object
102 // -------------------------------------------------------------------------------
103 alglib::minlpcreate(variable_num, Problem);
104 alglib::minlpsetlc2(Problem, constraint_general, lb_general, ub_general, constrant_num);
105 //alglib::minlpsetalgoipm(Problem);
106 alglib::minlpsetalgodss(Problem, 0.);
107 return Problem;
108}

110void agent::end_user::storage_schedule_LP_optimize(int foresight_time, sorted_vector expected_price_sorted, storage_inform &result, bool fixed_end){
111 int variable_num = 5 * foresight_time + 1;
112
113 // -------------------------------------------------------------------------------
114 // Set bounds for box constraints
115 // -------------------------------------------------------------------------------
116 Eigen::MatrixXd bound_box(variable_num, 2);
117 bound_box.row(0) << result.soc_ini, result.soc_ini;
118 for(int tick = 0; tick < foresight_time; ++ tick){
119 int q_self_ID = 5 * tick + 1;
120 int q_sell_ID = 5 * tick + 2;
121 int q_dc_ID = 5 * tick + 3;
122 int q_ch_ID = 5 * tick + 4;
123 int s_ID = 5 * tick + 5;
124
125 bound_box.row(q_self_ID) << 0, result.capacity_scale;
126 bound_box.row(q_sell_ID) << 0, result.capacity_scale;
127 bound_box.row(q_dc_ID) << 0, result.capacity_scale;
128 bound_box.row(q_ch_ID) << 0, result.capacity_scale;
129 bound_box.row(s_ID) << 0, result.energy_scale;
130 }
131 bound_box.bottomRows(1) << fixed_end * result.soc_final, fixed_end * result.soc_final + (1 - fixed_end) * result.energy_scale;
132
133 // Bounds of box constraints
134 alglib::real_1d_array lb_box;
135 alglib::real_1d_array ub_box;
136 lb_box.setcontent(bound_box.rows(), bound_box.col(0).data());
137 ub_box.setcontent(bound_box.rows(), bound_box.col(1).data());
138 alglib::minlpsetbc(result.Problem, lb_box, ub_box);
139
140 // -------------------------------------------------------------------------------
141 // Set objective coefficients of variables
142 // -------------------------------------------------------------------------------
143 Eigen::VectorXd obj_vec = Eigen::VectorXd::Zero(variable_num);
144 for(int tick = 0; tick < foresight_time; ++ tick){
145 int tick_ID = expected_price_sorted.id[tick];
146 int q_self_ID = 5 * tick_ID + 1;
147 int q_sell_ID = 5 * tick_ID + 2;
148 int q_ch_ID = 5 * tick_ID + 4;
149
150 obj_vec(q_self_ID) = -expected_price_sorted.value[tick] * result.efficiency;
151 obj_vec(q_sell_ID) = -expected_price_sorted.value[tick] * result.efficiency;
152 obj_vec(q_ch_ID) = expected_price_sorted.value[tick] / result.efficiency;
153 }
154 alglib::real_1d_array obj_coeff;
155 obj_coeff.setcontent(obj_vec.size(), obj_vec.data());
156 alglib::minlpsetcost(result.Problem, obj_coeff);
157
158 // -------------------------------------------------------------------------------
159 // Solve the problem and store the results
160 // -------------------------------------------------------------------------------
161 alglib::minlpoptimize(result.Problem);
162 alglib::real_1d_array sol;
163 alglib::minlpreport rep;
164 alglib::minlpresults(result.Problem, sol, rep);
165 Eigen::VectorXd sol_vec = Eigen::Map <Eigen::VectorXd> (sol.getcontent(), sol.length());
166
167 result.normalized_scheduled_capacity_profile = Eigen::VectorXd::Zero(foresight_time);
168 result.normalized_scheduled_soc_profile = Eigen::VectorXd::Zero(foresight_time);
169 for(int tick = 0; tick < foresight_time; ++ tick){
170 int q_dc_ID = 5 * tick + 3;
171 int q_ch_ID = 5 * tick + 4;
172 int s_ID = 5 * tick + 5;
173
174 result.normalized_scheduled_capacity_profile(tick) = sol_vec(q_dc_ID) - sol_vec(q_ch_ID);
175 result.normalized_scheduled_soc_profile(tick) = sol_vec(s_ID);
176 //std::cout << result.normalized_scheduled_capacity_profile.transpose() << "\n\n";
177 }
