8 void cbt_data_input(power_network::network_inform &Power_network_inform, std::string fin_cbt, std::string fin_entry){
9 // Read cross-border transmission data
10 auto fin_cbt_dim = basic::get_file_dim(fin_cbt);
11 auto fin_entry_dim = basic::get_file_dim(fin_entry);
12
13 // Store bidding zone names
14 Power_network_inform.cbt.bz_names = basic::get_col_name(fin_cbt, fin_cbt_dim[1]);
15
16 // Store entry points
17 Power_network_inform.cbt.entry_nodes = basic::read_file(fin_entry_dim[0], fin_entry_dim[1], fin_entry);
18 Power_network_inform.cbt.entry_nodes = Power_network_inform.cbt.entry_nodes.rightCols(fin_entry_dim[1] - 1);
19 Power_network_inform.cbt.entry_nodes = Power_network_inform.cbt.entry_nodes.array().max(0) - 1.;
20
21 // Find number of entry nodes for each bidding zones
22 Power_network_inform.cbt.entry_node_num = Eigen::VectorXi::Zero(fin_entry_dim[0]);
23 for(int zone_iter = 0; zone_iter < fin_entry_dim[0]; ++ zone_iter){
24 for(int node_iter = 0; node_iter < fin_entry_dim[1] - 1; ++ node_iter){
25 if(Power_network_inform.cbt.entry_nodes(zone_iter, node_iter) < 0.){
26 break;
27 }
28 Power_network_inform.cbt.entry_node_num(zone_iter) += 1;
29 }
30 }
31
32 // Store cbt constraints
33 Power_network_inform.cbt.flow_constraint = basic::read_file(fin_cbt_dim[0], fin_cbt_dim[1], fin_cbt);
34 }

164// Find the distance and covariance between points
165void power_network::point_distance_cov(points &point, double lambda){
166 double pi = boost::math::constants::pi<double>();
167 double tol = 1E-8;
168 Eigen::MatrixXd covariance = Eigen::MatrixXd::Ones(point.lon.size(), point.lon.size());
169 point.distance = Eigen::MatrixXd::Zero(point.lon.size(), point.lon.size());
170
171 #pragma omp parallel
172 {
173 #pragma omp for
174 for(int row_iter = 0; row_iter < point.lon.size() - 1; ++ row_iter){
175 for(int col_iter = row_iter + 1; col_iter < point.lon.size(); ++ col_iter){
176 point.distance(row_iter, col_iter) = spatial_field::geodist(Eigen::Vector2d(point.lon(row_iter) * pi / 180., point.lat(row_iter) * pi / 180.), Eigen::Vector2d(point.lon(col_iter) * pi / 180., point.lat(col_iter) * pi / 180.));
177 point.distance(col_iter, row_iter) = point.distance(row_iter, col_iter);
178 covariance(row_iter, col_iter) = exp(-pow(point.distance(row_iter, col_iter), 1.) / point.grid_length / lambda);
179 covariance(col_iter, row_iter) = covariance(row_iter, col_iter);
180 }
181 }
182 }
183
184 point.covariance = covariance.sparseView(tol);
185}
