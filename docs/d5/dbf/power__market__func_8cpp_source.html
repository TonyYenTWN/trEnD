16void power_market::Market_clearing_nodal(int tick, market_inform &Market, Eigen::VectorXi &default_price_ID, Eigen::MatrixXd &bidded_supply, Eigen::MatrixXd &bidded_demand){
17 Eigen::VectorXi price_demand_ID = (Market.price_intervals + 1) * Eigen::VectorXi::Ones(Market.num_zone);
18 Eigen::VectorXi price_supply_ID = Eigen::VectorXi::Zero(Market.num_zone);
19
20 //#pragma omp parallel for
21 for(int zone_ID = 0; zone_ID < Market.num_zone; ++ zone_ID){
22 while(price_demand_ID(zone_ID) > price_supply_ID(zone_ID)){
23 // Check if there are demand bids at current price interval
24 while(bidded_demand(price_demand_ID(zone_ID), zone_ID) == 0.){
25 if(price_demand_ID(zone_ID) > 0){
26 price_demand_ID(zone_ID) -= 1;
27 }
28 else{
29 // No available buyer left to buy electricity
30 default_price_ID(zone_ID) = price_supply_ID(zone_ID);
31 break;
32 }
33 }
34
35 // Check if there are supply bids at current price interval
36 while(bidded_supply(price_supply_ID(zone_ID), zone_ID) == 0.){
37 if(price_supply_ID(zone_ID) < Market.bidded_price.size() - 1){
38 price_supply_ID(zone_ID) += 1;
39 }
40 else{
41 // No available seller left to sell electricity
42 default_price_ID(zone_ID) = price_demand_ID(zone_ID);
43 break;
44 }
45 }
46
47 if(price_demand_ID(zone_ID) > price_supply_ID(zone_ID)){
48 double trade_quantity = std::min(bidded_supply(price_supply_ID(zone_ID), zone_ID), bidded_demand(price_demand_ID(zone_ID), zone_ID));
49 Market.confirmed_supply(tick, zone_ID) += trade_quantity;
50 Market.confirmed_demand(tick, zone_ID) += trade_quantity;
51 bidded_supply(price_supply_ID(zone_ID), zone_ID) -= trade_quantity;
52 bidded_demand(price_demand_ID(zone_ID), zone_ID) -= trade_quantity;
53 }
54 else{
55 if(bidded_supply(price_supply_ID(zone_ID), zone_ID) > 0){
56 default_price_ID(zone_ID) = price_supply_ID(zone_ID);
57 }
58 else{
59 default_price_ID(zone_ID) = price_demand_ID(zone_ID);
60 }
61 }
62 }
63
64 // Record default market clearing prices
65 Market.confirmed_price(tick, zone_ID) = Market.bidded_price(default_price_ID(zone_ID));
66 }
67}

138void power_market::Flow_Based_Market_LP_Set(market_inform &Market, alglib::minlpstate &Problem){
139 // -------------------------------------------------------------------------------
140 // LP Solver initialization for flow-based market optimization
141 // Warm-up once and reuse for the rest of time slices
142 // Variables are sorted as {V}, {S}, {S}_0, {S}_1, {S}_2, ..., {I}
143 // {S}_i is the set of source / sink at node #i at different price levels
144 // -------------------------------------------------------------------------------
145
146 // -------------------------------------------------------------------------------
147 // Set matrix for general constraints
148 // -------------------------------------------------------------------------------
149 // Construct node admittance matrix
150 double tol = 1E-10;
151 std::vector <Eigen::TripletXd> Y_n_trip;
152 Y_n_trip.reserve(2 * Market.network.num_edges + Market.network.num_vertice);
153 Eigen::SparseMatrix <double, Eigen::RowMajor> Y_n(Market.network.num_vertice, Market.network.num_vertice);
154 Eigen::VectorXd Y_n_diag = Eigen::VectorXd::Zero(Market.network.num_vertice);
155 Eigen::VectorXpd Connection_num = Eigen::VectorXpd::Ones(Market.network.num_vertice);
156 for(int edge_iter = 0; edge_iter < Market.network.num_edges; ++ edge_iter){
157 double y_edge;
158 y_edge = std::max(tol, abs(Market.network.admittance_vector(edge_iter)));
159 y_edge *= (Market.network.admittance_vector(edge_iter) > 0.);
160 // Equality constraints of voltage - source / sink at the nodes, off-diagonal terms
161 if(abs(y_edge) > tol){
162 Y_n_trip.push_back(Eigen::TripletXd(Market.network.incidence_matrix(edge_iter, 0), Market.network.incidence_matrix(edge_iter, 1), -y_edge));
163 Y_n_trip.push_back(Eigen::TripletXd(Market.network.incidence_matrix(edge_iter, 1), Market.network.incidence_matrix(edge_iter, 0), -y_edge));
164 Connection_num(Market.network.incidence_matrix(edge_iter, 0)) += 1;
165 Connection_num(Market.network.incidence_matrix(edge_iter, 1)) += 1;
166 }
167
168 // Equality constraints of voltage - source / sink at the nodes, diagonal terms
169 Y_n_diag(Market.network.incidence_matrix(edge_iter, 0)) += Market.network.admittance_vector(edge_iter);
170 Y_n_diag(Market.network.incidence_matrix(edge_iter, 1)) += Market.network.admittance_vector(edge_iter);
171 }
172 for(int node_iter = 0; node_iter < Market.network.num_vertice; ++ node_iter){
173 // Equality constraints of voltage - source / sink at the nodes, summed diagonal terms
174 Y_n_trip.push_back(Eigen::TripletXd(node_iter, node_iter, Y_n_diag(node_iter)));
175 }
176 Y_n.setFromTriplets(Y_n_trip.begin(), Y_n_trip.end());
177
178 // Generate sparse matrix for general (equality) constraints of voltage, power flow, and source / sink summation
179 int constrant_num = 2 * Market.network.num_vertice + Market.network.num_edges + 1;
180 int variable_num = Market.network.num_vertice * (Market.price_intervals + 4) + Market.network.num_edges;
181 Eigen::VectorXpd non_zero_num(constrant_num);
182 non_zero_num << Connection_num + Eigen::VectorXpd::Ones(Market.network.num_vertice), Eigen::VectorXpd::Constant(Market.network.num_edges, 3), Eigen::VectorXpd::Constant(Market.network.num_vertice, Market.price_intervals + 3), 1;
183 alglib::integer_1d_array row_sizes_general;
184 row_sizes_general.setcontent(non_zero_num.size(), non_zero_num.data());
185 alglib::sparsematrix constraint_general;
186 alglib::sparsecreatecrs(constrant_num, variable_num, row_sizes_general, constraint_general);
187 // Rows for voltage - source / sink equalities
188 for(int row_iter = 0; row_iter < Y_n.outerSize(); ++ row_iter){
189 for(Eigen::SparseMatrix<double, Eigen::RowMajor>::InnerIterator inner_iter(Y_n, row_iter); inner_iter; ++ inner_iter){
190 alglib::sparseset(constraint_general, inner_iter.row(), inner_iter.col(), inner_iter.value());
191 }
192 // Update the columns right to the node admittance block
193 alglib::sparseset(constraint_general, row_iter, Market.network.num_vertice + row_iter, -1.);
194 }
195 // Rows for voltage - power flow equalities
196 for(int edge_iter = 0; edge_iter < Market.network.num_edges; ++ edge_iter){
197 double y_edge;
198 y_edge = std::max(tol, abs(Market.network.admittance_vector(edge_iter)));
199 y_edge *= (y_edge > tol);
200 y_edge *= (Market.network.admittance_vector(edge_iter) > 0.);
201
202 if(Market.network.incidence_matrix(edge_iter, 0) < Market.network.incidence_matrix(edge_iter, 1)){
203 alglib::sparseset(constraint_general, Y_n.rows() + edge_iter, Market.network.incidence_matrix(edge_iter, 0), y_edge);
204 alglib::sparseset(constraint_general, Y_n.rows() + edge_iter, Market.network.incidence_matrix(edge_iter, 1), -y_edge);
205 }
206 else{
207 alglib::sparseset(constraint_general, Y_n.rows() + edge_iter, Market.network.incidence_matrix(edge_iter, 1), -y_edge);
208 alglib::sparseset(constraint_general, Y_n.rows() + edge_iter, Market.network.incidence_matrix(edge_iter, 0), y_edge);
209 }
210 alglib::sparseset(constraint_general, Y_n.rows() + edge_iter, Market.network.num_vertice * (Market.price_intervals + 4) + edge_iter, -1.);
211 }
212
213 // Rows for source / sink summation at each node
214 for(int node_iter = 0; node_iter < Y_n.rows(); ++ node_iter){
215 int row_ID = Y_n.rows() + Market.network.num_edges + node_iter;
216 alglib::sparseset(constraint_general, row_ID, Y_n.rows() + node_iter, 1.);
217 int col_start = 2 * Y_n.rows() + node_iter * (Market.price_intervals + 2);
218 int col_end = 2 * Y_n.rows() + (node_iter + 1) * (Market.price_intervals + 2) - 1;
219 for(int col_iter = col_start; col_iter <= col_end; ++ col_iter){
220 alglib::sparseset(constraint_general, row_ID, col_iter, -1.);
221 }
222 }
223
224 // Voltage at reference node
225 alglib::sparseset(constraint_general, non_zero_num.size() - 1, 0, 1.);
226
227 // Check if the sparse matrix is correct
228// std::cout << Y_n << "\n\n";
229// double value;
230// for(int row_iter = 0; row_iter < non_zero_num.size(); ++ row_iter){
231// for(int col_iter = 0; col_iter < 2 * Y_n.cols() + 3; ++ col_iter){
232// value = sparseget(constraint_general, row_iter, col_iter);
233// std::cout << value << "\t";
234// }
235// std::cout << "\t";
236// for(int col_iter = Market.network.num_vertice * (Market.price_intervals + 4); col_iter < Market.network.num_vertice * (Market.price_intervals + 4) + 1; ++ col_iter){
237// value = sparseget(constraint_general, row_iter, col_iter);
238// std::cout << value << "\t";
239// }
240// std::cout << "\n";
241// }
242
243 // -------------------------------------------------------------------------------
244 // Set bounds for general and box constraints
245 // -------------------------------------------------------------------------------
246 Eigen::MatrixXd bound_general = Eigen::MatrixXd::Zero(constrant_num, 2);
247 Eigen::MatrixXd bound_box(variable_num, 2);
248 std::cout << variable_num << "\n";
249 bound_box.topRows(Market.network.num_vertice) = Market.network.voltage_constraint;
250 bound_box.middleRows(Market.network.num_vertice, Market.network.num_vertice).col(0) = Eigen::VectorXd::Constant(Market.network.num_vertice, -std::numeric_limits<double>::infinity());
251 bound_box.middleRows(Market.network.num_vertice, Market.network.num_vertice).col(1) = Eigen::VectorXd::Constant(Market.network.num_vertice, std::numeric_limits<double>::infinity());
252 bound_box.bottomRows(Market.network.num_edges) = Market.network.power_constraint;
253
254 // Bounds of general constraints
255 alglib::real_1d_array lb_general;
256 alglib::real_1d_array ub_general;
257 lb_general.setcontent(bound_general.rows(), bound_general.col(0).data());
258 ub_general.setcontent(bound_general.rows(), bound_general.col(1).data());
259
260 // Bounds of box constraints
261 alglib::real_1d_array lb_box;
262 alglib::real_1d_array ub_box;
263 lb_box.setcontent(bound_box.rows(), bound_box.col(0).data());
264 ub_box.setcontent(bound_box.rows(), bound_box.col(1).data());
265
266 // -------------------------------------------------------------------------------
267 // Set scale of variables
268 // -------------------------------------------------------------------------------
269 Eigen::VectorXd scale_vec = Eigen::VectorXd::Ones(variable_num);
270 scale_vec.head(Market.network.num_vertice) = bound_box.col(1).head(Market.network.num_vertice) - bound_box.col(0).head(Market.network.num_vertice);
271 scale_vec.tail(Market.network.num_edges) = bound_box.col(1).tail(Market.network.num_edges) - bound_box.col(0).tail(Market.network.num_edges);
272 alglib::real_1d_array scale;
273 scale.setcontent(scale_vec.size(), scale_vec.data());
274
275 // -------------------------------------------------------------------------------
276 // Set objective coefficients of variables
277 // -------------------------------------------------------------------------------
278 Eigen::VectorXd obj_vec = Eigen::VectorXd::Zero(variable_num);
279 for(int node_iter = 0; node_iter < Market.network.num_vertice; ++ node_iter){
280 int row_start = 2 * Market.network.num_vertice + node_iter * (Market.price_intervals + 2);
281 obj_vec.segment(row_start, Market.price_intervals + 2) = Market.bidded_price;
282 }
283 alglib::real_1d_array obj_coeff;
284 obj_coeff.setcontent(obj_vec.size(), obj_vec.data());
285
286 // -------------------------------------------------------------------------------
287 // Set the LP problem object
288 // -------------------------------------------------------------------------------
289 alglib::minlpcreate(variable_num, Problem);
290 alglib::minlpsetcost(Problem, obj_coeff);
291// alglib::minlpsetbc(Problem, lb_box, ub_box);
292 alglib::minlpsetlc2(Problem, constraint_general, lb_general, ub_general, constrant_num);
293 alglib::minlpsetscale(Problem, scale);
294 //alglib::minlpsetalgoipm(Problem);
295 alglib::minlpsetalgodss(Problem, 0.);

299void power_market::Flow_Based_Market_Optimization(int tick, market_inform &Market, alglib::minlpstate &Problem){
300 // -------------------------------------------------------------------------------
301 // Update bounds for box constraints
302 // -------------------------------------------------------------------------------
303 int variable_num = Market.network.num_vertice * (Market.price_intervals + 4) + Market.network.num_edges;
304 Eigen::MatrixXd bound_box(variable_num, 2);
305 bound_box.topRows(Market.network.num_vertice) = Market.network.voltage_constraint;
306 bound_box.middleRows(Market.network.num_vertice, Market.network.num_vertice).col(0) = Eigen::VectorXd::Constant(Market.network.num_vertice, -std::numeric_limits<double>::infinity());
307 bound_box.middleRows(Market.network.num_vertice, Market.network.num_vertice).col(1) = Eigen::VectorXd::Constant(Market.network.num_vertice, std::numeric_limits<double>::infinity());
308 for(int node_iter = 0; node_iter < Market.network.num_vertice; ++ node_iter){
309 int row_start = 2 * Market.network.num_vertice + node_iter * (Market.price_intervals + 2);
310 bound_box.middleRows(row_start, Market.price_intervals + 2).col(0) = -Market.submitted_demand.col(node_iter);
311 bound_box.middleRows(row_start, Market.price_intervals + 2).col(1) = Market.submitted_supply.col(node_iter);
312 }
313 bound_box.bottomRows(Market.network.num_edges) = Market.network.power_constraint;
314
315 // Bounds of box constraints
316 alglib::real_1d_array lb_box;
317 alglib::real_1d_array ub_box;
318 lb_box.setcontent(bound_box.rows(), bound_box.col(0).data());
319 ub_box.setcontent(bound_box.rows(), bound_box.col(1).data());
320 alglib::minlpsetbc(Market.Problem, lb_box, ub_box);
321
322 // -------------------------------------------------------------------------------
323 // Solve the problem
324 // -------------------------------------------------------------------------------
325 alglib::real_1d_array sol;
326 alglib::minlpreport rep;
327 alglib::minlpoptimize(Market.Problem);
328 alglib::minlpresults(Market.Problem, sol, rep);
329
330 // -------------------------------------------------------------------------------
331 // Store the solution
332 // -------------------------------------------------------------------------------
333 Eigen::VectorXd sol_vec = Eigen::Map <Eigen::VectorXd> (sol.getcontent(), bound_box.rows());
334 for(int node_iter = 0; node_iter < Market.network.num_vertice; ++ node_iter){
335 // Store power source / sink
336 int row_start = 2 * Market.network.num_vertice + node_iter * (Market.price_intervals + 2);
337 Market.confirmed_supply(tick, node_iter) = (sol_vec.segment(row_start, Market.price_intervals + 2).array().max(0)).sum();
338 Market.confirmed_demand(tick, node_iter) = -(sol_vec.segment(row_start, Market.price_intervals + 2).array().min(0)).sum();
339 Market.confirmed_price(tick, node_iter) = Market.bidded_price(0) + rep.lagbc[row_start];
340 Market.confirmed_price(tick, node_iter) = std::min(Market.confirmed_price(tick, node_iter), Market.price_range_inflex(1));
341 Market.confirmed_price(tick, node_iter) = std::max(Market.confirmed_price(tick, node_iter), Market.price_range_inflex(0));
342
343 // Store voltage and power flow
344 Market.network.confirmed_voltage.row(tick) = sol_vec.head(Market.network.num_vertice);
345 Market.network.confirmed_power.row(tick) = sol_vec.tail(Market.network.num_edges);
346 }

348
