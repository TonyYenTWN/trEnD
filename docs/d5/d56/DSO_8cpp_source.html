10 double pi = boost::math::constants::pi<double>();
11 DSO_Markets.clear();
12 DSO_Markets = std::vector <market_inform> (Power_network_inform.DSO_cluster.size());
13
14 // Initialize markets at each clustered DSO
15 for(int DSO_iter = 0; DSO_iter < DSO_Markets.size(); ++ DSO_iter){
16 // Input parameters of DSO market
17 DSO_Markets[DSO_iter].num_zone = Power_network_inform.DSO_cluster[DSO_iter].points_ID.size() + Power_network_inform.DSO_cluster[DSO_iter].nodes_ID.size();
18 DSO_Markets[DSO_iter].time_intervals = Time;
19 DSO_Markets[DSO_iter].set_bidded_price();
20
21 // Set compact incidence matrix and edge admittance matrix
22 // Use fractional Laplacian here :
23 // y(x_1, x_2) = N / 4 / pi / z / |x_1 - x_2|^(2 + d), where
24 // y(x_1, x_2): admittance between x_1 and x_2
25 // N: mean line density at the neighborhood of x_1 containing also x_2
26 // z: per length impedence of power lines
27 // d: fractional dimension of the power line distribution
28 DSO_Markets[DSO_iter].network.num_vertice = DSO_Markets[DSO_iter].num_zone;
29 if(DSO_Markets[DSO_iter].num_zone > 1){
30 DSO_Markets[DSO_iter].network.num_edges = DSO_Markets[DSO_iter].num_zone * (DSO_Markets[DSO_iter].num_zone - 1) / 2;
31 DSO_Markets[DSO_iter].network.incidence_matrix = Eigen::MatrixXi(DSO_Markets[DSO_iter].network.num_edges, 2);
32 DSO_Markets[DSO_iter].network.admittance_vector = Eigen::VectorXd(DSO_Markets[DSO_iter].network.num_edges);
33 int edge_iter = 0;
34 for(int row_iter = 0; row_iter < DSO_Markets[DSO_iter].network.num_vertice - 1; ++ row_iter){
35 for(int col_iter = row_iter + 1; col_iter < DSO_Markets[DSO_iter].network.num_vertice; ++ col_iter){
36 DSO_Markets[DSO_iter].network.incidence_matrix(edge_iter, 0) = row_iter;
37 DSO_Markets[DSO_iter].network.incidence_matrix(edge_iter, 1) = col_iter;
38 DSO_Markets[DSO_iter].network.admittance_vector(edge_iter) = Power_network_inform.tech_parameters.line_density_distr / 4. / pi;
39 DSO_Markets[DSO_iter].network.admittance_vector(edge_iter) /= Power_network_inform.tech_parameters.z_distr_series.imag() * 1E3;
40 double distance;
41
42 if(row_iter < Power_network_inform.DSO_cluster[DSO_iter].points_ID.size() && col_iter < Power_network_inform.DSO_cluster[DSO_iter].points_ID.size()){
43 int point_ID_1 = Power_network_inform.DSO_cluster[DSO_iter].points_ID[row_iter];
44 int point_ID_2 = Power_network_inform.DSO_cluster[DSO_iter].points_ID[col_iter];
45 distance = Power_network_inform.points.distance(point_ID_1, point_ID_2);
46 }
47 else{
48 if(row_iter < Power_network_inform.DSO_cluster[DSO_iter].points_ID.size()){
49 int point_ID = Power_network_inform.DSO_cluster[DSO_iter].points_ID[row_iter];
50 int node_ID = Power_network_inform.DSO_cluster[DSO_iter].nodes_ID[col_iter - Power_network_inform.DSO_cluster[DSO_iter].points_ID.size()];
51 Eigen::Vector2d point_coor = Eigen::Vector2d(Power_network_inform.points.lon(point_ID), Power_network_inform.points.lat(point_ID));
52 Eigen::Vector2d node_coor = Eigen::Vector2d(Power_network_inform.nodes.lon(point_ID), Power_network_inform.nodes.lat(point_ID));
53 point_coor *= pi / 180.;
54 node_coor *= pi / 180.;
55 distance = spatial_field::geodist(point_coor, node_coor);
56 }
57 else{
58 int node_ID_1 = Power_network_inform.DSO_cluster[DSO_iter].nodes_ID[row_iter - Power_network_inform.DSO_cluster[DSO_iter].points_ID.size()];
59 int node_ID_2 = Power_network_inform.DSO_cluster[DSO_iter].nodes_ID[col_iter - Power_network_inform.DSO_cluster[DSO_iter].points_ID.size()];
60 Eigen::Vector2d node_coor_1 = Eigen::Vector2d(Power_network_inform.points.lon(node_ID_1), Power_network_inform.points.lat(node_ID_1));
61 Eigen::Vector2d node_coor_2 = Eigen::Vector2d(Power_network_inform.nodes.lon(node_ID_2), Power_network_inform.nodes.lat(node_ID_2));
62 node_coor_1 *= pi / 180.;
63 node_coor_2 *= pi / 180.;
64 distance = spatial_field::geodist(node_coor_1, node_coor_2);
65 }
66 }
67 DSO_Markets[DSO_iter].network.admittance_vector(edge_iter) /= pow(distance * 1E-3, 2. + Power_network_inform.tech_parameters.fraction_dim_distr);
68 edge_iter += 1;
69 }
70 }
71 }
72
73 // Set voltage and power constraints at each edge
74 DSO_Markets[DSO_iter].network.voltage_constraint = Eigen::MatrixXd::Ones(DSO_Markets[DSO_iter].network.num_vertice, 2);
75 DSO_Markets[DSO_iter].network.voltage_constraint.col(0) *= -pi / 12;
76 DSO_Markets[DSO_iter].network.voltage_constraint.col(1) *= pi / 12;
77 DSO_Markets[DSO_iter].network.power_constraint = Eigen::MatrixXd::Ones(DSO_Markets[DSO_iter].network.num_edges, 2);
78 DSO_Markets[DSO_iter].network.power_constraint.col(0) *= -50.;
79 DSO_Markets[DSO_iter].network.power_constraint.col(1) *= 50.;
80
81 // Initialization of process variables
82 power_market::Market_Initialization(DSO_Markets[DSO_iter]);
83
84 // Initialization of output variables
85 DSO_Markets[DSO_iter].confirmed_supply = Eigen::MatrixXd::Zero(Time, DSO_Markets[DSO_iter].num_zone);
86 DSO_Markets[DSO_iter].confirmed_demand = Eigen::MatrixXd::Zero(Time, DSO_Markets[DSO_iter].num_zone);
87 DSO_Markets[DSO_iter].confirmed_price = Eigen::MatrixXd(Time, DSO_Markets[DSO_iter].num_zone);
88 DSO_Markets[DSO_iter].network.confirmed_power = Eigen::MatrixXd(Time, DSO_Markets[DSO_iter].network.num_edges);
89 DSO_Markets[DSO_iter].network.confirmed_voltage = Eigen::MatrixXd(Time, DSO_Markets[DSO_iter].network.num_vertice);
90 }
91}
