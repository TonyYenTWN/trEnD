8 double pi = boost::math::constants::pi<double>();
9 DSO_Markets.clear();

12 // Initialize markets at each clustered DSO
13 for(int DSO_iter = 0; DSO_iter < DSO_Markets.size(); ++ DSO_iter){
14 // Input parameters of DSO market
15 DSO_Markets[DSO_iter].num_zone = Power_network_inform.DSO_cluster[DSO_iter].points_ID.size() + Power_network_inform.DSO_cluster[DSO_iter].nodes_ID.size();
16 DSO_Markets[DSO_iter].time_intervals = Time;
17 //DSO_Markets[DSO_iter].set_bidded_price();
18 parameters::bidded_price(DSO_Markets[DSO_iter].bidded_price_map);

20 // Set node admittance matrix
21 // Use fractional Laplacian here :
22 // y(x_1, x_2) = N / 4 / pi / z / |x_1 - x_2|^(2 + d), where
23 // y(x_1, x_2): admittance between x_1 and x_2
24 // N: mean line density at the neighborhood of x_1 containing also x_2
25 // z: per length impedence of power lines
26 // d: fractional dimension of the power line distribution
27 double z_base_low = pow(Power_network_inform.tech_parameters.voltage_cutoff_distr, 2.) / Power_network_inform.tech_parameters.s_base / 3.;
28 double z_base_high = pow(Power_network_inform.tech_parameters.voltage_cutoff_connection, 2.) / Power_network_inform.tech_parameters.s_base / 3.;
29 double partition_func = 0.;
30 DSO_Markets[DSO_iter].network.num_vertice = DSO_Markets[DSO_iter].num_zone;
31 Eigen::MatrixXd admittance = Eigen::MatrixXd::Ones(DSO_Markets[DSO_iter].network.num_vertice, DSO_Markets[DSO_iter].network.num_vertice);
32 Eigen::MatrixXd distance = Eigen::MatrixXd::Zero(DSO_Markets[DSO_iter].network.num_vertice, DSO_Markets[DSO_iter].network.num_vertice);

37 for(int row_iter = 0; row_iter < DSO_Markets[DSO_iter].network.num_vertice - 1; ++ row_iter){
38 for(int col_iter = row_iter + 1; col_iter < DSO_Markets[DSO_iter].network.num_vertice; ++ col_iter){
39 if(row_iter < Power_network_inform.DSO_cluster[DSO_iter].points_ID.size() && col_iter < Power_network_inform.DSO_cluster[DSO_iter].points_ID.size()){
40 int point_ID_1 = Power_network_inform.DSO_cluster[DSO_iter].points_ID[row_iter];
41 int point_ID_2 = Power_network_inform.DSO_cluster[DSO_iter].points_ID[col_iter];
42 distance(row_iter, col_iter) = Power_network_inform.points.distance(point_ID_1, point_ID_2);
43 admittance(row_iter, col_iter) = 1. / pow(distance(row_iter, col_iter) * 1E-3, 1. + Power_network_inform.tech_parameters.fraction_dim_distr);
44 partition_func += admittance(row_iter, col_iter);
45 }
46 else{
47 if(row_iter < Power_network_inform.DSO_cluster[DSO_iter].points_ID.size()){

52 point_coor *= pi / 180.;
53 node_coor *= pi / 180.;
54 distance(row_iter, col_iter) = spatial_field::geodist(point_coor, node_coor);
55 }
56 else{
57 int node_ID_1 = Power_network_inform.DSO_cluster[DSO_iter].nodes_ID[row_iter - Power_network_inform.DSO_cluster[DSO_iter].points_ID.size()];
58 int node_ID_2 = Power_network_inform.DSO_cluster[DSO_iter].nodes_ID[col_iter - Power_network_inform.DSO_cluster[DSO_iter].points_ID.size()];
59 Eigen::Vector2d node_coor_1 = Eigen::Vector2d(Power_network_inform.nodes.lon(node_ID_1), Power_network_inform.nodes.lat(node_ID_1));

61 node_coor_1 *= pi / 180.;
62 node_coor_2 *= pi / 180.;
63 distance(row_iter, col_iter) = spatial_field::geodist(node_coor_1, node_coor_2);
64 }
65 }
66 }
67 }
68 admittance.topLeftCorner(Power_network_inform.DSO_cluster[DSO_iter].points_ID.size(), Power_network_inform.DSO_cluster[DSO_iter].points_ID.size()) *= Power_network_inform.tech_parameters.line_density_distr * Power_network_inform.DSO_cluster[DSO_iter].points_ID.size() * z_base_low;
69 admittance.topLeftCorner(Power_network_inform.DSO_cluster[DSO_iter].points_ID.size(), Power_network_inform.DSO_cluster[DSO_iter].points_ID.size()) /= partition_func * Power_network_inform.tech_parameters.z_distr_series.imag();
70 admittance.rightCols(Power_network_inform.DSO_cluster[DSO_iter].nodes_ID.size()) *= z_base_high / Power_network_inform.tech_parameters.z_distr_series.imag();
71 int demo = std::min(15, DSO_Markets[DSO_iter].network.num_vertice);
72 admittance = admittance.array() / distance.array();
73
74 // Set compact incidence matrix and edge admittance matrix
75 double tol = 1.;
76 DSO_Markets[DSO_iter].network.incidence.reserve(DSO_Markets[DSO_iter].network.num_vertice * DSO_Markets[DSO_iter].network.num_vertice);
77 DSO_Markets[DSO_iter].network.admittance.reserve(DSO_Markets[DSO_iter].network.num_vertice * DSO_Markets[DSO_iter].network.num_vertice);
78 std::vector <double> power_limit;
79 power_limit.reserve(DSO_Markets[DSO_iter].network.num_vertice * DSO_Markets[DSO_iter].network.num_vertice);
80
81 for(int row_iter = 0; row_iter < DSO_Markets[DSO_iter].network.num_vertice - 1; ++ row_iter){
82 for(int col_iter = row_iter + 1; col_iter < DSO_Markets[DSO_iter].network.num_vertice; ++ col_iter){
83 if(row_iter < Power_network_inform.DSO_cluster[DSO_iter].points_ID.size() && col_iter < Power_network_inform.DSO_cluster[DSO_iter].points_ID.size()){
84 if(admittance(row_iter, col_iter) > tol){
85 DSO_Markets[DSO_iter].network.incidence.push_back(Eigen::Vector2i(row_iter, col_iter));
86 DSO_Markets[DSO_iter].network.admittance.push_back(admittance(row_iter , col_iter));
87 power_limit.push_back(Power_network_inform.tech_parameters.voltage_cutoff_distr);
88 }
89 }
90 else{
91 DSO_Markets[DSO_iter].network.incidence.push_back(Eigen::Vector2i(row_iter, col_iter));
92 DSO_Markets[DSO_iter].network.admittance.push_back(admittance(row_iter, col_iter));
93 power_limit.push_back(Power_network_inform.tech_parameters.voltage_cutoff_connection);
94 }
95 }
96 }
97 DSO_Markets[DSO_iter].network.num_edges = DSO_Markets[DSO_iter].network.incidence.size();
98
99 // Set voltage and power constraints at each edge
100 DSO_Markets[DSO_iter].network.voltage_constraint = Eigen::MatrixXd::Ones(DSO_Markets[DSO_iter].network.num_vertice, 2);
101 DSO_Markets[DSO_iter].network.voltage_constraint.col(0) *= -Power_network_inform.tech_parameters.theta_limit;
102 DSO_Markets[DSO_iter].network.voltage_constraint.col(1) *= Power_network_inform.tech_parameters.theta_limit;
103 DSO_Markets[DSO_iter].network.power_constraint = Eigen::MatrixXd (DSO_Markets[DSO_iter].network.num_edges, 2);
104 DSO_Markets[DSO_iter].network.power_constraint.col(1) = Eigen::Map <Eigen::VectorXd> (power_limit.data(), power_limit.size());
105 DSO_Markets[DSO_iter].network.power_constraint.col(1) /= Power_network_inform.tech_parameters.s_base;
106 DSO_Markets[DSO_iter].network.power_constraint.col(0) = -DSO_Markets[DSO_iter].network.power_constraint.col(1);
107
108 // Initialization of process variables
109 power_market::Market_Initialization(DSO_Markets[DSO_iter]);
110
111 // Initialization of output variables
112 DSO_Markets[DSO_iter].confirmed_supply = Eigen::MatrixXd::Zero(Time, DSO_Markets[DSO_iter].num_zone);
113 DSO_Markets[DSO_iter].confirmed_demand = Eigen::MatrixXd::Zero(Time, DSO_Markets[DSO_iter].num_zone);
114 DSO_Markets[DSO_iter].confirmed_price = Eigen::MatrixXd::Zero(Time, DSO_Markets[DSO_iter].num_zone);
115 DSO_Markets[DSO_iter].confirmed_ratio_supply = Eigen::MatrixXd::Zero(Time, DSO_Markets[DSO_iter].num_zone);
116 DSO_Markets[DSO_iter].confirmed_ratio_demand = Eigen::MatrixXd::Zero(Time, DSO_Markets[DSO_iter].num_zone);

358void power_market::DSO_Market_Results_Get(int tick, market_inform &Market, alglib::minlpstate &Problem, power_network::DSO_cluster &DSO_cluster, bool supply){
359 alglib::real_1d_array sol;
360 alglib::minlpreport rep;
361 alglib::minlpresults(Problem, sol, rep);
362 Eigen::VectorXd sol_vec = Eigen::Map <Eigen::VectorXd> (sol.getcontent(), sol.length());
363
364 if(supply){
365 // Store filtered supply bids at spatial points
366 for(int point_iter = 0; point_iter < DSO_cluster.points_ID.size(); ++ point_iter){
367 // Store power sink
368 int row_start = 2 * Market.network.num_vertice + point_iter * (Market.price_intervals + 2);
369 Market.confirmed_supply(tick, point_iter) = (sol_vec.segment(row_start, Market.price_intervals + 2).array().max(0)).sum();
370
371 // Store nodal prices
372 Market.confirmed_price(tick, point_iter) = Market.bidded_price_map.bidded_price(1) + rep.lagbc[row_start + 1];
373 Market.confirmed_price(tick, point_iter) = int(Market.confirmed_price(tick, point_iter)) + .5;
374 if(Market.confirmed_price(tick, point_iter) < Market.bidded_price_map.bidded_price(1)){
375 Market.confirmed_price(tick, point_iter) = Market.bidded_price_map.bidded_price(0);
376 }
377 else if(Market.confirmed_price(tick, point_iter) > Market.bidded_price_map.bidded_price(Market.price_intervals)){
378 Market.confirmed_price(tick, point_iter) = Market.bidded_price_map.bidded_price(Market.price_intervals + 1);
379 }
380
381 // Store ratio at nodes
382 for(int price_iter = 0; price_iter < Market.price_intervals + 2; ++ price_iter){
383 if(Market.bidded_price_map.bidded_price(price_iter) >= Market.confirmed_price(tick, point_iter) || price_iter == Market.price_intervals + 1){
384 if(sol[row_start + price_iter] >= 0.){
385 Market.confirmed_ratio_demand(point_iter) = std::min(Market.submitted_demand(price_iter, point_iter), Market.submitted_supply(price_iter, point_iter) - sol[row_start + price_iter]);
386 Market.confirmed_ratio_supply(point_iter) = Market.confirmed_ratio_demand(point_iter) + sol[row_start + price_iter];
387 Market.confirmed_ratio_demand(point_iter) /= Market.submitted_demand(price_iter, point_iter) + 1E-12;
388 }
389 else{
390 Market.confirmed_ratio_supply(point_iter) = std::min(Market.submitted_supply(price_iter, point_iter), Market.submitted_demand(price_iter, point_iter) + sol[row_start + price_iter]);
391 Market.confirmed_ratio_supply(point_iter) /= Market.submitted_supply(price_iter, point_iter) + 1E-12;
392 }
393 break;
394 }
395 }
396 }
397 }
398 else{
399 // Store filtered demand bids at spatial points
400 for(int point_iter = 0; point_iter < DSO_cluster.points_ID.size(); ++ point_iter){
401 // Store power sink
402 int row_start = 2 * Market.network.num_vertice + point_iter * (Market.price_intervals + 2);
403 Market.confirmed_demand(tick, point_iter) = -(sol_vec.segment(row_start, Market.price_intervals + 2).array().min(0)).sum();
404
405 // Store nodal prices
406 Market.confirmed_price(tick, point_iter) = Market.bidded_price_map.bidded_price(1) + rep.lagbc[row_start + 1];
407 Market.confirmed_price(tick, point_iter) = int(Market.confirmed_price(tick, point_iter)) + .5;
408 if(Market.confirmed_price(tick, point_iter) < Market.bidded_price_map.bidded_price(1)){
409 Market.confirmed_price(tick, point_iter) = Market.bidded_price_map.bidded_price(0);
410 }
411 else if(Market.confirmed_price(tick, point_iter) > Market.bidded_price_map.bidded_price(Market.price_intervals)){
412 Market.confirmed_price(tick, point_iter) = Market.bidded_price_map.bidded_price(Market.price_intervals + 1);
413 }
414
415 // Store ratio at nodes
416 for(int price_iter = 0; price_iter < Market.price_intervals + 2; ++ price_iter){
417 if(Market.bidded_price_map.bidded_price(price_iter) >= Market.confirmed_price(tick, point_iter) || price_iter == Market.price_intervals + 1){
418 if(sol[row_start + price_iter] >= 0.){

428 }
