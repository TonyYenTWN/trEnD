7void power_market::DSO_Markets_Set(markets_inform &DSO_Markets, power_network::network_inform &Power_network_inform, int Time){
8 double pi = boost::math::constants::pi<double>();
9 DSO_Markets.clear();
10 DSO_Markets = std::vector <market_inform> (Power_network_inform.DSO_cluster.size());
11
12 // Initialize markets at each clustered DSO
13 for(int DSO_iter = 0; DSO_iter < DSO_Markets.size(); ++ DSO_iter){
14 // Input parameters of DSO market
15 DSO_Markets[DSO_iter].num_zone = Power_network_inform.DSO_cluster[DSO_iter].points_ID.size() + Power_network_inform.DSO_cluster[DSO_iter].nodes_ID.size();
16 DSO_Markets[DSO_iter].time_intervals = Time;
17 DSO_Markets[DSO_iter].set_bidded_price();
18
19 // Set node admittance matrix
20 // Use fractional Laplacian here :
21 // y(x_1, x_2) = N / 4 / pi / z / |x_1 - x_2|^(2 + d), where
22 // y(x_1, x_2): admittance between x_1 and x_2
23 // N: mean line density at the neighborhood of x_1 containing also x_2
24 // z: per length impedence of power lines
25 // d: fractional dimension of the power line distribution
26 double z_base_low = pow(Power_network_inform.tech_parameters.voltage_cutoff_distr, 2.) / Power_network_inform.tech_parameters.s_base / 3.;
27 double z_base_high = pow(Power_network_inform.tech_parameters.voltage_cutoff_connection, 2.) / Power_network_inform.tech_parameters.s_base / 3.;
28 double partition_func = 0.;
29 DSO_Markets[DSO_iter].network.num_vertice = DSO_Markets[DSO_iter].num_zone;
30 Eigen::MatrixXd admittance = Eigen::MatrixXd::Ones(DSO_Markets[DSO_iter].network.num_vertice, DSO_Markets[DSO_iter].network.num_vertice);
31 Eigen::MatrixXd distance = Eigen::MatrixXd::Zero(DSO_Markets[DSO_iter].network.num_vertice, DSO_Markets[DSO_iter].network.num_vertice);
32
33// std::cout << "-------------------------------------------------------------------------------------------------------------------------------------------------\n";
34// std::cout << DSO_iter << "\n";
35// std::cout << "-------------------------------------------------------------------------------------------------------------------------------------------------\n";
36 for(int row_iter = 0; row_iter < DSO_Markets[DSO_iter].network.num_vertice - 1; ++ row_iter){
37 for(int col_iter = row_iter + 1; col_iter < DSO_Markets[DSO_iter].network.num_vertice; ++ col_iter){
38 if(row_iter < Power_network_inform.DSO_cluster[DSO_iter].points_ID.size() && col_iter < Power_network_inform.DSO_cluster[DSO_iter].points_ID.size()){
39 int point_ID_1 = Power_network_inform.DSO_cluster[DSO_iter].points_ID[row_iter];
40 int point_ID_2 = Power_network_inform.DSO_cluster[DSO_iter].points_ID[col_iter];
41 distance(row_iter, col_iter) = Power_network_inform.points.distance(point_ID_1, point_ID_2);
42 admittance(row_iter, col_iter) = 1. / pow(distance(row_iter, col_iter) * 1E-3, 1. + Power_network_inform.tech_parameters.fraction_dim_distr);
43 partition_func += admittance(row_iter, col_iter);
44 }
45 else{
46 if(row_iter < Power_network_inform.DSO_cluster[DSO_iter].points_ID.size()){
47 int point_ID = Power_network_inform.DSO_cluster[DSO_iter].points_ID[row_iter];
48 int node_ID = Power_network_inform.DSO_cluster[DSO_iter].nodes_ID[col_iter - Power_network_inform.DSO_cluster[DSO_iter].points_ID.size()];
49 Eigen::Vector2d point_coor = Eigen::Vector2d(Power_network_inform.points.lon(point_ID), Power_network_inform.points.lat(point_ID));
50 Eigen::Vector2d node_coor = Eigen::Vector2d(Power_network_inform.nodes.lon(node_ID), Power_network_inform.nodes.lat(node_ID));
51 point_coor *= pi / 180.;
52 node_coor *= pi / 180.;
53 distance(row_iter, col_iter) = spatial_field::geodist(point_coor, node_coor);
54 }
55 else{
56 int node_ID_1 = Power_network_inform.DSO_cluster[DSO_iter].nodes_ID[row_iter - Power_network_inform.DSO_cluster[DSO_iter].points_ID.size()];
57 int node_ID_2 = Power_network_inform.DSO_cluster[DSO_iter].nodes_ID[col_iter - Power_network_inform.DSO_cluster[DSO_iter].points_ID.size()];
58 Eigen::Vector2d node_coor_1 = Eigen::Vector2d(Power_network_inform.nodes.lon(node_ID_1), Power_network_inform.nodes.lat(node_ID_1));
59 Eigen::Vector2d node_coor_2 = Eigen::Vector2d(Power_network_inform.nodes.lon(node_ID_2), Power_network_inform.nodes.lat(node_ID_2));
60 node_coor_1 *= pi / 180.;
61 node_coor_2 *= pi / 180.;
62 distance(row_iter, col_iter) = spatial_field::geodist(node_coor_1, node_coor_2);
63 }
64 }
65 }
66 }
67 admittance.topLeftCorner(Power_network_inform.DSO_cluster[DSO_iter].points_ID.size(), Power_network_inform.DSO_cluster[DSO_iter].points_ID.size()) *= Power_network_inform.tech_parameters.line_density_distr * Power_network_inform.DSO_cluster[DSO_iter].points_ID.size() * z_base_low;
68 admittance.topLeftCorner(Power_network_inform.DSO_cluster[DSO_iter].points_ID.size(), Power_network_inform.DSO_cluster[DSO_iter].points_ID.size()) /= partition_func * Power_network_inform.tech_parameters.z_distr_series.imag();
69 admittance.rightCols(Power_network_inform.DSO_cluster[DSO_iter].nodes_ID.size()) *= z_base_high / Power_network_inform.tech_parameters.z_distr_series.imag();
70 int demo = std::min(15, DSO_Markets[DSO_iter].network.num_vertice);
71 admittance = admittance.array() / distance.array();
72
73 // Set compact incidence matrix and edge admittance matrix
74 double tol = 1.;
75 DSO_Markets[DSO_iter].network.incidence.reserve(DSO_Markets[DSO_iter].network.num_vertice * DSO_Markets[DSO_iter].network.num_vertice);
76 DSO_Markets[DSO_iter].network.admittance.reserve(DSO_Markets[DSO_iter].network.num_vertice * DSO_Markets[DSO_iter].network.num_vertice);
77 std::vector <double> power_limit;
78 power_limit.reserve(DSO_Markets[DSO_iter].network.num_vertice * DSO_Markets[DSO_iter].network.num_vertice);
79
80 for(int row_iter = 0; row_iter < DSO_Markets[DSO_iter].network.num_vertice - 1; ++ row_iter){
81 for(int col_iter = row_iter + 1; col_iter < DSO_Markets[DSO_iter].network.num_vertice; ++ col_iter){
82 if(row_iter < Power_network_inform.DSO_cluster[DSO_iter].points_ID.size() && col_iter < Power_network_inform.DSO_cluster[DSO_iter].points_ID.size()){
83 if(admittance(row_iter, col_iter) > tol){
84 DSO_Markets[DSO_iter].network.incidence.push_back(Eigen::Vector2i(row_iter, col_iter));
85 DSO_Markets[DSO_iter].network.admittance.push_back(admittance(row_iter , col_iter));
86 power_limit.push_back(Power_network_inform.tech_parameters.voltage_cutoff_distr);
87 }
88 }
89 else{
90 DSO_Markets[DSO_iter].network.incidence.push_back(Eigen::Vector2i(row_iter, col_iter));
91 DSO_Markets[DSO_iter].network.admittance.push_back(admittance(row_iter , col_iter));
92 power_limit.push_back(Power_network_inform.tech_parameters.voltage_cutoff_connection);
93 }
94 }
95 }
96 DSO_Markets[DSO_iter].network.num_edges = DSO_Markets[DSO_iter].network.incidence.size();
97
98 // Set voltage and power constraints at each edge
99 DSO_Markets[DSO_iter].network.voltage_constraint = Eigen::MatrixXd::Ones(DSO_Markets[DSO_iter].network.num_vertice, 2);
100 DSO_Markets[DSO_iter].network.voltage_constraint.col(0) *= -Power_network_inform.tech_parameters.theta_limit;
101 DSO_Markets[DSO_iter].network.voltage_constraint.col(1) *= Power_network_inform.tech_parameters.theta_limit;
102 DSO_Markets[DSO_iter].network.power_constraint = Eigen::MatrixXd (DSO_Markets[DSO_iter].network.num_edges, 2);
103 DSO_Markets[DSO_iter].network.power_constraint.col(1) = Eigen::Map <Eigen::VectorXd> (power_limit.data(), power_limit.size());
104 DSO_Markets[DSO_iter].network.power_constraint.col(1) /= Power_network_inform.tech_parameters.s_base;
105 DSO_Markets[DSO_iter].network.power_constraint.col(0) = -DSO_Markets[DSO_iter].network.power_constraint.col(1);
106
107 // Initialization of process variables
108 power_market::Market_Initialization(DSO_Markets[DSO_iter]);
109
110 // Initialization of output variables
111 DSO_Markets[DSO_iter].confirmed_supply = Eigen::MatrixXd::Zero(Time, DSO_Markets[DSO_iter].num_zone);
112 DSO_Markets[DSO_iter].confirmed_demand = Eigen::MatrixXd::Zero(Time, DSO_Markets[DSO_iter].num_zone);
113 DSO_Markets[DSO_iter].confirmed_price = Eigen::MatrixXd(Time, DSO_Markets[DSO_iter].num_zone);
114 DSO_Markets[DSO_iter].network.confirmed_power = Eigen::MatrixXd(Time, DSO_Markets[DSO_iter].network.num_edges);
115 DSO_Markets[DSO_iter].network.confirmed_voltage = Eigen::MatrixXd(Time, DSO_Markets[DSO_iter].network.num_vertice);
116 }

184 for(int point_iter = 0; point_iter < end_user_profiles.size(); ++ point_iter){
185 int node_ID = Power_network_inform.points.node(point_iter);
186 int sample_num = agent::parameters::sample_num();
187 double node_price = TSO_Market.actual_price(tick - 1, node_ID);

190 end_user_profiles[point_iter][sample_iter].operation.smart_appliance.unfulfilled_demand = 0.;
191 end_user_profiles[point_iter][sample_iter].operation.EV.BESS.soc_ini -= end_user_profiles[point_iter][sample_iter].operation.EV.BESS.normalized_scheduled_capacity_profile(0);
192 end_user_profiles[point_iter][sample_iter].operation.BESS.soc_ini -= end_user_profiles[point_iter][sample_iter].operation.BESS.normalized_scheduled_capacity_profile(0);
193
194 // Demand settlement
195 // Order of reduction: BESS charge, EV charge, smart appliance, inflexible demand
196 double marginal_demand = 0.;
197 marginal_demand += (node_price == end_user_profiles[point_iter][sample_iter].operation.demand_flex_price) * (end_user_profiles[point_iter][sample_iter].operation.normalized_scheduled_residual_demand_flex_profile(0) >= 0.) * end_user_profiles[point_iter][sample_iter].operation.normalized_scheduled_residual_demand_flex_profile(0);
198 marginal_demand += (node_price == end_user_profiles[point_iter][sample_iter].operation.demand_inflex_price) * (end_user_profiles[point_iter][sample_iter].operation.normalized_scheduled_residual_demand_inflex_profile(0) >= 0.) * end_user_profiles[point_iter][sample_iter].operation.normalized_scheduled_residual_demand_inflex_profile(0);
199 double demand_gap = (1. - TSO_Market.actual_price_ratio(tick - 1, node_ID)) * marginal_demand;
200 demand_gap += (node_price > end_user_profiles[point_iter][sample_iter].operation.demand_flex_price) * (end_user_profiles[point_iter][sample_iter].operation.normalized_scheduled_residual_demand_flex_profile(0) >= 0.) * end_user_profiles[point_iter][sample_iter].operation.normalized_scheduled_residual_demand_flex_profile(0);

238 end_user_profiles[point_iter][sample_iter].operation.BESS.soc_ini += reduction;
