12 DSO_Markets = std::vector <market_inform> (Power_network_inform.DSO_cluster.size());

14 // Initialize markets at each clustered DSO
15 for(int DSO_iter = 0; DSO_iter < DSO_Markets.size(); ++ DSO_iter){
16 // Input parameters of DSO market
17 DSO_Markets[DSO_iter].num_zone = Power_network_inform.DSO_cluster[DSO_iter].points_ID.size() + Power_network_inform.DSO_cluster[DSO_iter].nodes_ID.size();
18 DSO_Markets[DSO_iter].time_intervals = Time;
19 DSO_Markets[DSO_iter].set_bidded_price();
20
21 // Set node admittance matrix
22 // Use fractional Laplacian here :
23 // y(x_1, x_2) = N / 4 / pi / z / |x_1 - x_2|^(2 + d), where
24 // y(x_1, x_2): admittance between x_1 and x_2
25 // N: mean line density at the neighborhood of x_1 containing also x_2
26 // z: per length impedence of power lines
27 // d: fractional dimension of the power line distribution
28 double z_base_low = pow(Power_network_inform.tech_parameters.voltage_cutoff_distr, 2.) / Power_network_inform.tech_parameters.s_base / 3.;
29 double z_base_high = pow(Power_network_inform.tech_parameters.voltage_cutoff_connection, 2.) / Power_network_inform.tech_parameters.s_base / 3.;
30 double partition_func = 0.;
31 DSO_Markets[DSO_iter].network.num_vertice = DSO_Markets[DSO_iter].num_zone;
32 Eigen::MatrixXd admittance = Eigen::MatrixXd::Ones(DSO_Markets[DSO_iter].network.num_vertice, DSO_Markets[DSO_iter].network.num_vertice);
33 Eigen::MatrixXd distance = Eigen::MatrixXd::Zero(DSO_Markets[DSO_iter].network.num_vertice, DSO_Markets[DSO_iter].network.num_vertice);
34
35// std::cout << "-------------------------------------------------------------------------------------------------------------------------------------------------\n";
36// std::cout << DSO_iter << "\n";
37// std::cout << "-------------------------------------------------------------------------------------------------------------------------------------------------\n";
38 for(int row_iter = 0; row_iter < DSO_Markets[DSO_iter].network.num_vertice - 1; ++ row_iter){
39 for(int col_iter = row_iter + 1; col_iter < DSO_Markets[DSO_iter].network.num_vertice; ++ col_iter){
40 if(row_iter < Power_network_inform.DSO_cluster[DSO_iter].points_ID.size() && col_iter < Power_network_inform.DSO_cluster[DSO_iter].points_ID.size()){
41 int point_ID_1 = Power_network_inform.DSO_cluster[DSO_iter].points_ID[row_iter];
42 int point_ID_2 = Power_network_inform.DSO_cluster[DSO_iter].points_ID[col_iter];
43 distance(row_iter, col_iter) = Power_network_inform.points.distance(point_ID_1, point_ID_2);
44 admittance(row_iter, col_iter) = 1. / pow(distance(row_iter, col_iter) * 1E-3, 1. + Power_network_inform.tech_parameters.fraction_dim_distr);
45 partition_func += admittance(row_iter, col_iter);
46 }
47 else{
48 if(row_iter < Power_network_inform.DSO_cluster[DSO_iter].points_ID.size()){
49 int point_ID = Power_network_inform.DSO_cluster[DSO_iter].points_ID[row_iter];
50 int node_ID = Power_network_inform.DSO_cluster[DSO_iter].nodes_ID[col_iter - Power_network_inform.DSO_cluster[DSO_iter].points_ID.size()];
51 Eigen::Vector2d point_coor = Eigen::Vector2d(Power_network_inform.points.lon(point_ID), Power_network_inform.points.lat(point_ID));
52 Eigen::Vector2d node_coor = Eigen::Vector2d(Power_network_inform.nodes.lon(node_ID), Power_network_inform.nodes.lat(node_ID));
53 point_coor *= pi / 180.;
54 node_coor *= pi / 180.;
55 distance(row_iter, col_iter) = spatial_field::geodist(point_coor, node_coor);
56 }
57 else{
58 int node_ID_1 = Power_network_inform.DSO_cluster[DSO_iter].nodes_ID[row_iter - Power_network_inform.DSO_cluster[DSO_iter].points_ID.size()];
59 int node_ID_2 = Power_network_inform.DSO_cluster[DSO_iter].nodes_ID[col_iter - Power_network_inform.DSO_cluster[DSO_iter].points_ID.size()];
60 Eigen::Vector2d node_coor_1 = Eigen::Vector2d(Power_network_inform.nodes.lon(node_ID_1), Power_network_inform.nodes.lat(node_ID_1));
61 Eigen::Vector2d node_coor_2 = Eigen::Vector2d(Power_network_inform.nodes.lon(node_ID_2), Power_network_inform.nodes.lat(node_ID_2));
62 node_coor_1 *= pi / 180.;
63 node_coor_2 *= pi / 180.;
64 distance(row_iter, col_iter) = spatial_field::geodist(node_coor_1, node_coor_2);
65 }
66 }
67 }
68 }
69 admittance.topLeftCorner(Power_network_inform.DSO_cluster[DSO_iter].points_ID.size(), Power_network_inform.DSO_cluster[DSO_iter].points_ID.size()) *= Power_network_inform.tech_parameters.line_density_distr * Power_network_inform.DSO_cluster[DSO_iter].points_ID.size() * z_base_low;
70 admittance.topLeftCorner(Power_network_inform.DSO_cluster[DSO_iter].points_ID.size(), Power_network_inform.DSO_cluster[DSO_iter].points_ID.size()) /= partition_func * Power_network_inform.tech_parameters.z_distr_series.imag();
71 admittance.rightCols(Power_network_inform.DSO_cluster[DSO_iter].nodes_ID.size()) *= z_base_high / Power_network_inform.tech_parameters.z_distr_series.imag();
72 int demo = std::min(15, DSO_Markets[DSO_iter].network.num_vertice);
73 admittance = admittance.array() / distance.array();
74
75 // Set compact incidence matrix and edge admittance matrix
76 double tol = 1.;
77 DSO_Markets[DSO_iter].network.incidence.reserve(DSO_Markets[DSO_iter].network.num_vertice * DSO_Markets[DSO_iter].network.num_vertice);
78 DSO_Markets[DSO_iter].network.admittance.reserve(DSO_Markets[DSO_iter].network.num_vertice * DSO_Markets[DSO_iter].network.num_vertice);
79 std::vector <double> power_limit;
80 power_limit.reserve(DSO_Markets[DSO_iter].network.num_vertice * DSO_Markets[DSO_iter].network.num_vertice);
81
82 for(int row_iter = 0; row_iter < DSO_Markets[DSO_iter].network.num_vertice - 1; ++ row_iter){
83 for(int col_iter = row_iter + 1; col_iter < DSO_Markets[DSO_iter].network.num_vertice; ++ col_iter){
84 if(row_iter < Power_network_inform.DSO_cluster[DSO_iter].points_ID.size() && col_iter < Power_network_inform.DSO_cluster[DSO_iter].points_ID.size()){
85 if(admittance(row_iter, col_iter) > tol){
86 DSO_Markets[DSO_iter].network.incidence.push_back(Eigen::Vector2i(row_iter, col_iter));
87 DSO_Markets[DSO_iter].network.admittance.push_back(admittance(row_iter , col_iter));
88 power_limit.push_back(Power_network_inform.tech_parameters.voltage_cutoff_distr);
89 }
90 }
91 else{
92 DSO_Markets[DSO_iter].network.incidence.push_back(Eigen::Vector2i(row_iter, col_iter));
93 DSO_Markets[DSO_iter].network.admittance.push_back(admittance(row_iter , col_iter));
94 power_limit.push_back(Power_network_inform.tech_parameters.voltage_cutoff_connection);
95 }
96 }
97 }
98 DSO_Markets[DSO_iter].network.num_edges = DSO_Markets[DSO_iter].network.incidence.size();
99
100 // Set voltage and power constraints at each edge
101 DSO_Markets[DSO_iter].network.voltage_constraint = Eigen::MatrixXd::Ones(DSO_Markets[DSO_iter].network.num_vertice, 2);
102 DSO_Markets[DSO_iter].network.voltage_constraint.col(0) *= -Power_network_inform.tech_parameters.theta_limit;
103 DSO_Markets[DSO_iter].network.voltage_constraint.col(1) *= Power_network_inform.tech_parameters.theta_limit;
104 DSO_Markets[DSO_iter].network.power_constraint = Eigen::MatrixXd (DSO_Markets[DSO_iter].network.num_edges, 2);
105 DSO_Markets[DSO_iter].network.power_constraint.col(1) = Eigen::Map <Eigen::VectorXd> (power_limit.data(), power_limit.size());
106 DSO_Markets[DSO_iter].network.power_constraint.col(1) /= Power_network_inform.tech_parameters.s_base;
107 DSO_Markets[DSO_iter].network.power_constraint.col(0) = -DSO_Markets[DSO_iter].network.power_constraint.col(1);
108
109 // Initialization of process variables
110 power_market::Market_Initialization(DSO_Markets[DSO_iter]);
111
112 // Initialization of output variables
113 DSO_Markets[DSO_iter].confirmed_supply = Eigen::MatrixXd::Zero(Time, DSO_Markets[DSO_iter].num_zone);
114 DSO_Markets[DSO_iter].confirmed_demand = Eigen::MatrixXd::Zero(Time, DSO_Markets[DSO_iter].num_zone);
115 DSO_Markets[DSO_iter].confirmed_price = Eigen::MatrixXd(Time, DSO_Markets[DSO_iter].num_zone);
116 DSO_Markets[DSO_iter].network.confirmed_power = Eigen::MatrixXd(Time, DSO_Markets[DSO_iter].network.num_edges);
117 DSO_Markets[DSO_iter].network.confirmed_voltage = Eigen::MatrixXd(Time, DSO_Markets[DSO_iter].network.num_vertice);
118 }
