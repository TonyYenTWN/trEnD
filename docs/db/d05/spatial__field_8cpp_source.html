4void spatial_field::BME_copula(inference_inform &inform, power_network::network_inform &Power_network_inform, Eigen::SparseMatrix <double> &Constraint, double tol){
5 int bz_num = Constraint.cols();
6 int point_num = Constraint.rows();
7
8 // Iterations
9 int count = 0;
10 Eigen::VectorXd dmu = Eigen::VectorXd::Constant(point_num, 1.);
11 while(count < 5000 && dmu.lpNorm<Eigen::Infinity>() > tol){
12 // Matrices for the linear system
13 Eigen::SparseMatrix <double> Conversion_Mat_1(point_num, point_num);
14 std::vector <Eigen::TripletXd> Conversion_Mat_1_Trip;
15 Conversion_Mat_1.reserve(point_num);
16
17 for(int item = 0; item < point_num; ++ item){
18 double diff_x = .5 * inform.mu_scale(item) * pow((-log(1. - cdf(inform.norm_dist, inform.x(item)))), -.5) * pow(1. - cdf(inform.norm_dist, inform.x(item)), -1.) * pdf(inform.norm_dist, inform.x(item));
19 Conversion_Mat_1_Trip.push_back(Eigen::TripletXd(item, item, diff_x));
20 }
21 Conversion_Mat_1.setFromTriplets(Conversion_Mat_1_Trip.begin(), Conversion_Mat_1_Trip.end());
22
23 Eigen::SparseMatrix <double> Conversion_Mat_2 = Conversion_Mat_1 * Constraint;
24
25 // Solve the linear system
26 Eigen::SparseLU <Eigen::SparseMatrix <double>, Eigen::COLAMDOrdering<int>> solver;
27 solver.compute(Conversion_Mat_2.transpose() * Power_network_inform.points.covariance * Conversion_Mat_2);
28 Eigen::VectorXd lambda = solver.solve(inform.mu_mean - Constraint.transpose() * inform.mu + Conversion_Mat_2.transpose() * inform.x);
29 Eigen::VectorXd dx = Power_network_inform.points.covariance * Conversion_Mat_2 * lambda - inform.x;
30 inform.x += inform.alpha_iteration * dx;
31
32 // Update the solution
33 for(int item = 0; item < point_num; ++ item){
34 dmu(item) = inform.mu_scale(item) * pow(-log(1. - cdf(inform.norm_dist, inform.x(item))), .5) - inform.mu(item);
35 inform.mu(item) += dmu(item);
36 }
37 Eigen::VectorXd mu_inv_0 = pow(inform.mu.array(), -1.);
38 dmu *= mu_inv_0;
39
40 count += 1;
41 }
42}

47void spatial_field::nominal_demand_inference(power_network::network_inform &Power_network_inform){
48 int bz_num = Power_network_inform.points.bidding_zone.maxCoeff() + 1;
49 int point_num = Power_network_inform.points.bidding_zone.size();
50 int Time = power_market::parameters::Time();

52
53 // Read electricity demand data
54 std::string fin_demand = "csv/input/spatial_field/demand_actual_2021.csv";
55 auto fin_demand_dim = basic::get_file_dim(fin_demand);
56 auto Demand_ts = basic::read_file(fin_demand_dim[0], fin_demand_dim[1], fin_demand);
57
58 // Set sparse matrix for equality constaints
59 Eigen::SparseMatrix <double> Constraint (point_num, bz_num);
60 std::vector<Eigen::TripletXd> Constraint_Trip;
61 Constraint_Trip.reserve(point_num);
62 for(int point_iter = 0; point_iter < point_num; ++ point_iter){
63 int bz_ID = Power_network_inform.points.bidding_zone(point_iter);
64 double value = Power_network_inform.points.population_density(point_iter) * Power_network_inform.points.point_area;
65 Constraint_Trip.push_back(Eigen::TripletXd(point_iter, bz_ID, value));
66 }
67
68 Constraint.setFromTriplets(Constraint_Trip.begin(), Constraint_Trip.end());
69
70 // ------------------------------------------------------------------------------------------------------------------------------------------
71 // Infer the annual average of the normalized mean demand field
72 // ------------------------------------------------------------------------------------------------------------------------------------------
73 // Initialization of parameters
74 inference_inform nominal_demand;
75 nominal_demand.alpha_iteration = 1.;
76 nominal_demand.mu_mean = Demand_ts.colwise().sum().tail(bz_num);
77 nominal_demand.mu_mean /= Time;
78 // Mean of weibull distribution = mu_0_scale * gamma(1 + 1 / k), here k = 2
79 double mu_0_scale = nominal_demand.mu_mean.sum() / Constraint.sum() * 2. / pow(pi, .5);
80 nominal_demand.mu_scale = Eigen::VectorXd::Constant(point_num, mu_0_scale);
81 nominal_demand.mu = Eigen::VectorXd::Constant(point_num, mu_0_scale);
82 nominal_demand.x = Eigen::VectorXd(point_num);
83 for(int item = 0; item < point_num; ++ item){
84 nominal_demand.x(item) = quantile(nominal_demand.norm_dist, 1. - exp(-pow(nominal_demand.mu(item) / nominal_demand.mu_scale(item), 2.)));
85 }
86
87 // Inference step
88 BME_copula(nominal_demand, Power_network_inform, Constraint, 1E-12);
89
90 // Output the annual average of normalized mean demand field
91 std::string fout_name;
92 fout_name = "csv/processed/spatial_field/nominal_mean_demand_field_10km_annual_mean.csv";
93 std::vector <std::string> col_name;
94 col_name.push_back("nominal_mean_demand");
95 basic::write_file(nominal_demand.mu, fout_name, col_name);
96
97 // ------------------------------------------------------------------------------------------------------------------------------------------
98 // Infer the normalized mean demand field
99 // ------------------------------------------------------------------------------------------------------------------------------------------
100 // Initialization of parameters
101 nominal_demand.alpha_iteration = .01;
102 nominal_demand.mu_scale = nominal_demand.mu * 2. / pow(pi, .5);
103 nominal_demand.x_scale = Eigen::VectorXd(point_num);
104 for(int item = 0; item < point_num; ++ item){
105 nominal_demand.x_scale(item) = quantile(nominal_demand.norm_dist, 1. - exp(-pow(1., 2.)));
106 }
107 nominal_demand.mu = nominal_demand.mu_scale;
108 nominal_demand.x = nominal_demand.x_scale;
109
110 // Run the algorithm for each time slice
111 for(int tick = 0; tick < 1; ++ tick){
112 nominal_demand.mu_mean = Demand_ts.row(tick).tail(bz_num);
113
114 // Inference step
115 BME_copula(nominal_demand, Power_network_inform, Constraint, 1E-3);
116
117 // Output normalized mean demand field
118 int count_zeros = 0;
119 int tick_temp = tick;
120 std::string digit_zeros;
121 while(int (tick_temp / 10) != 0){
122 count_zeros += 1;
123 tick_temp /= 10;
124 }
125 for(int item = 0; item < 5 - count_zeros; ++item){
126 digit_zeros += std::to_string(0);
127 }
128 fout_name = "csv/processed/spatial_field/nominal_mean_demand_field_10km_ts_" + digit_zeros + std::to_string(tick) + ".csv";
129 basic::write_file(nominal_demand.mu, fout_name, col_name);
130 digit_zeros.clear();
131 }
