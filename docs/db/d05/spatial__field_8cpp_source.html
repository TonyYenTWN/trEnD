36 for(int item = 0; item < point_num; ++ item){
37 x_0(item) = quantile(norm_dist, 1. - exp(-pow(mu_0(item) / mu_0_scale, 2.)));

46 std::vector <Eigen::TripletXd> Conversion_Mat_1_Trip;

48
49 for(int item = 0; item < point_num; ++ item){

51 Conversion_Mat_1_Trip.push_back(Eigen::TripletXd(item, item, diff_x));
52 }

54

64 // Update the solution

66 dmu_0(item) = mu_0_scale * pow(-log(1. - cdf(norm_dist, x_0(item))), .5) - mu_0(item);

102 while(count < 5000 && dmu.lpNorm<Eigen::Infinity>() > pow(10., -3.)){

116 // Solve the linear system

118 solver.compute(Conversion_Mat_2.transpose() * Power_network_inform.points.covariance * Conversion_Mat_2);
119 Eigen::VectorXd lambda = solver.solve(mu_mean - Constraint.transpose() * mu + Conversion_Mat_2.transpose() * x);

152void spatial_field::nominal_demand_inference(power_network::network_inform &Power_network_inform){
153 int bz_num = Power_network_inform.points.bidding_zone.maxCoeff() + 1;
154 int point_num = Power_network_inform.points.bidding_zone.size();
155
156 // Read electricity demand data
157 std::string fin_demand = "csv/input/spatial_field/demand_actual_2021.csv";
158 auto fin_demand_dim = basic::get_file_dim(fin_demand);
159 auto Demand_ts = basic::read_file(fin_demand_dim[0], fin_demand_dim[1], fin_demand);
160
161 // Set sparse matrix for equality constaints
162 Eigen::SparseMatrix <double> Constraint (point_num, bz_num);
163 std::vector<Eigen::TripletXd> Constraint_Trip;
164 Constraint_Trip.reserve(point_num);
165
166 for(int point_iter = 0; point_iter < point_num; ++ point_iter){
167 int bz_ID = Power_network_inform.points.bidding_zone(point_iter);
168 double value = Power_network_inform.points.population_density(point_iter) * Power_network_inform.points.point_area;
169 Constraint_Trip.push_back(Eigen::TripletXd(point_iter, bz_ID, value));
170 }
171
172 Constraint.setFromTriplets(Constraint_Trip.begin(), Constraint_Trip.end());
173
174 // Spatial field inference
175 BME(Power_network_inform, Constraint, Demand_ts);
176}
