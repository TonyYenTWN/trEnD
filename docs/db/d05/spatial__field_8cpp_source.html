5 void BME_copula(spatial_field::inference_inform &inform, power_network::network_inform &Power_network_inform, Eigen::SparseMatrix <double> &Constraint, double tol){
6 int bz_num = Constraint.cols();
7 int point_num = Constraint.rows();
8
9 // Iterations
10 int count = 0;
11 Eigen::VectorXd dmu = Eigen::VectorXd::Constant(point_num, 1.);
12 while(count < 5000 && dmu.lpNorm<Eigen::Infinity>() > tol){
13 // Matrices for the linear system
14 Eigen::SparseMatrix <double> Conversion_Mat_1(point_num, point_num);
15 std::vector <Eigen::TripletXd> Conversion_Mat_1_Trip;
16 Conversion_Mat_1.reserve(point_num);
17
18 for(int item = 0; item < point_num; ++ item){
19 double diff_x = .5 * inform.mu_scale(item) * pow((-log(1. - cdf(inform.norm_dist, inform.x(item)))), -.5) * pow(1. - cdf(inform.norm_dist, inform.x(item)), -1.) * pdf(inform.norm_dist, inform.x(item));
20 Conversion_Mat_1_Trip.push_back(Eigen::TripletXd(item, item, diff_x));
21 }
22 Conversion_Mat_1.setFromTriplets(Conversion_Mat_1_Trip.begin(), Conversion_Mat_1_Trip.end());
23
24 Eigen::SparseMatrix <double> Conversion_Mat_2 = Conversion_Mat_1 * Constraint;
25
26 // Solve the linear system
27 Eigen::MatrixXd solver = Conversion_Mat_2.transpose() * Power_network_inform.points.covariance * Conversion_Mat_2;
28 Eigen::VectorXd lambda = solver.colPivHouseholderQr().solve(inform.mu_mean - Constraint.transpose() * inform.mu + Conversion_Mat_2.transpose() * inform.x);
29 Eigen::VectorXd dx = Power_network_inform.points.covariance * Conversion_Mat_2 * lambda - inform.x;
30 inform.x += inform.alpha_iteration * dx;
31
32 // Update the solution
33 for(int item = 0; item < point_num; ++ item){
34 dmu(item) = inform.mu_scale(item) * pow(-log(1. - cdf(inform.norm_dist, inform.x(item))), .5) - inform.mu(item);
35 inform.mu(item) += dmu(item);
36 }
37 Eigen::VectorXd mu_inv_0 = pow(inform.mu.array(), -1.);
38 dmu *= mu_inv_0;
39
40 //std::cout << inform.mu.minCoeff() << "\t" << inform.mu.maxCoeff() << "\t" << dmu.lpNorm<Eigen::Infinity>() << "\n";
41 count += 1;
42 }
43 //std::cout << inform.mu.minCoeff() << "\t" << inform.mu.maxCoeff() << "\t" << dmu.lpNorm<Eigen::Infinity>() << "\n";
44 }

46 void BME_linear(spatial_field::inference_inform &inform, Eigen::SparseMatrix <double> &Constraint){
47 // ---------------------------------------------------------------------
48 // MaxEnt, known covariance and mean = 0, reference equiprobable space = x
49 // COV^(-1) * x* - Constraint * lambda = 0
50 // x* = COV * Constraint * lambda
51 // Constraint^T * (x* + x_0) = Constraint^T * COV * Constraint * lambda + Constraint^T * x_0 = demand
52 // lambda = solve(Constraint^T * COV * Constraint, demand - Constraint^T * x_0)
53 // ---------------------------------------------------------------------
54 // Solve the linear system
55 Eigen::VectorXd Constraint_sum = Constraint.transpose() * inform.mu_scale;
56 Eigen::SparseLU <Eigen::SparseMatrix <double>, Eigen::COLAMDOrdering<int>> solver;
57 solver.compute(inform.Conversion_Mat_1);
58 Eigen::VectorXd lambda = solver.solve(inform.mu_mean - Constraint_sum);
59 inform.mu = inform.Conversion_Mat_2 * lambda;
60 inform.mu += inform.mu_scale;
61 //std::cout << inform.mu.minCoeff() << "\t" << inform.mu.maxCoeff() << "\n\n";
62 }

93 // Set sparse matrix for equality constraints for nominal demand

96 Constraint_demand_Trip.reserve(point_num);
97 for(int point_iter = 0; point_iter < point_num; ++ point_iter){

99 double value = Power_network_inform.points.population_density(point_iter) * Power_network_inform.points.point_area;
100 Constraint_demand_Trip.push_back(Eigen::TripletXd(point_iter, bz_ID, value));

103 //std::cout << Constraint_demand.topRows(10) << "\n\n";
104
105 // ------------------------------------------------------------------------------------------------------------------------------------------
106 // Infer the annual average of the normalized mean demand field
107 // ------------------------------------------------------------------------------------------------------------------------------------------
108 // Initialization of parameters
109 inference_inform nominal_demand;
110 nominal_demand.alpha_iteration = 1.;
111 nominal_demand.mu_mean = Demand_ts.colwise().sum().tail(bz_num);
112 nominal_demand.mu_mean /= Time;
113 // Mean of weibull distribution = mu_0_scale * gamma(1 + 1 / k), here k = 2
114 double mu_0_scale = nominal_demand.mu_mean.sum() / Constraint_demand.sum() * 2. / pow(pi, .5);
115 nominal_demand.mu_scale = Eigen::VectorXd::Constant(point_num, mu_0_scale);
116 nominal_demand.mu = Eigen::VectorXd::Constant(point_num, mu_0_scale);
117 nominal_demand.x = Eigen::VectorXd(point_num);
118 for(int item = 0; item < point_num; ++ item){
119 nominal_demand.x(item) = quantile(nominal_demand.norm_dist, 1. - exp(-pow(nominal_demand.mu(item) / nominal_demand.mu_scale(item), 2.)));
120 }
121
122 // Inference step
123 BME_copula(nominal_demand, Power_network_inform, Constraint_demand, 1E-12);

130 col_name.push_back("nominal_mean_demand");

137 Constraint_imbalance_Trip.reserve(point_num);
138 for(int point_iter = 0; point_iter < point_num; ++ point_iter){

166 basic::write_file(imbalance.mu, fout_name, col_name);

232 // Capacity factor fields
233 // ------------------------------------------------------------------------------------------------------------------------------------------
234 // Set sparse matrix for equality constraints for nominal demand
235 Eigen::MatrixXd Constraint_wind_on_dense = Eigen::MatrixXd::Zero(point_num, bz_num);

254 wind_on_cf.mu_mean = (Wind_on_ts.colwise().sum()).segment(fin_wind_on_row_name, bz_num);

283 wind_on_cf.mu = Eigen::VectorXd::Constant(point_num, mu_0_scale);
