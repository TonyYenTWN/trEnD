104 // Functions
105 /**Set the bidding prices of the power market.*/
106 void set_bidded_price(){
107 this->bidded_price = Eigen::VectorXd(this->price_intervals + 2);
108 this->bidded_price(0) = this->price_range_inflex(0);
109 this->bidded_price.array().tail(1) = this->price_range_inflex(1);
110 this->bidded_price.segment(1, this->price_intervals) = Eigen::VectorXd::LinSpaced(this->price_intervals, this->price_range_flex(0) + .5 * (this->price_range_flex(1) - this->price_range_flex(0)) / this->price_intervals, this->price_range_flex(1) - .5 * (this->price_range_flex(1) - this->price_range_flex(0)) / this->price_intervals);
111 }
