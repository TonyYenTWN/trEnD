130 // Functions
131 /**Set the bidding prices of the power market.*/
132 void set_bidded_price(){
133 this->bidded_price = Eigen::VectorXd(this->price_intervals + 2);
134 this->bidded_price(0) = this->price_range_inflex(0);
135 this->bidded_price.array().tail(1) = this->price_range_inflex(1);
136 this->bidded_price.segment(1, this->price_intervals) = Eigen::VectorXd::LinSpaced(this->price_intervals, this->price_range_flex(0) + .5 * (this->price_range_flex(1) - this->price_range_flex(0)) / this->price_intervals, this->price_range_flex(1) - .5 * (this->price_range_flex(1) - this->price_range_flex(0)) / this->price_intervals);
137 }
