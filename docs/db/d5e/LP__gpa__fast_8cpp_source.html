6// Delete redundant equality constraints
7void LP_constraint_eq_redundant_deletion(LP_object &Problem){
8 // Because an initial feasible point is needed for the algorithm, feasibility should be checked earlier
9 // However, redundant eqaulity constraints can still occur so this step should still be performed
10
11 Problem.Solver.qr.compute(Problem.Constraint.eq_orig_matrix.transpose());
12
13 // Check if the equality constraints has maximum possible rank
14 if(Problem.Solver.qr.rank() < Problem.Constraints_eq_num){
15 // Update the information about equality constraints after removing the redundant ones
16 Problem.Constraint.eq_orig_matrix = Problem.Solver.qr.colsPermutation() * Problem.Constraint.eq_orig_matrix;
17 Problem.Constraint.eq_orig_matrix = Problem.Constraint.eq_orig_matrix.topRows(Problem.Solver.qr.rank());
18 Problem.Constraints_eq_num = Problem.Solver.qr.rank();
19 Problem.Boundary.eq_vector = Problem.Solver.qr.colsPermutation() * Problem.Boundary.eq_vector;
20 Problem.Boundary.eq_vector = Problem.Boundary.eq_vector.head(Problem.Solver.qr.rank());
21 }
22}

24// Choose redundant variables forming full rank square matrix from the equality constraints
25void LP_variables_permutation(LP_object &Problem, bool stepwise_obj){
26 // Set precision of zero detection
27 double tol = pow(10, -12);
28
29 std::vector<Trip> Permutation_trip;
30 Permutation_trip.reserve(Problem.Variables_num);
31 int redundant_size = 0;
32 int remaining_size = 0;
33 int start_id;
34 Eigen::SparseMatrix <double> Redundant_matrix;
35 Eigen::SparseMatrix <double> Subspan_matrix(Problem.Variables_num, Problem.Constraints_eq_num);
36 Subspan_matrix.reserve(Eigen::VectorXi::Constant(Problem.Constraints_eq_num, 2));
37 Problem.Constraint.permutation_matrix = Eigen::SparseMatrix <double> (Problem.Variables_num, Problem.Variables_num);
38
39 // The initial column should be non-zero
40 for(int var_iter = 0; var_iter < Problem.Variables_num; ++ var_iter){
41 if(Problem.Constraint.eq_orig_matrix.col(var_iter).norm() > tol){
42 Subspan_matrix.insert(var_iter, redundant_size) = 1;
43 Permutation_trip.push_back(Trip(var_iter, Problem.Variables_num - Problem.Constraints_eq_num, 1));
44 start_id = var_iter + 1;
45 redundant_size += 1;
46 break;
47 }
48 else{
49 Permutation_trip.push_back(Trip(var_iter, remaining_size, 1));
50 remaining_size += 1;
51 }
52 }
53 for(int var_iter = start_id; var_iter < Problem.Variables_num; ++ var_iter){
54 if(redundant_size < Problem.Constraints_eq_num){
55 Subspan_matrix.insert(var_iter, redundant_size) = 1;
56 Redundant_matrix = Problem.Constraint.eq_orig_matrix * Subspan_matrix;
57 Problem.Solver.qr.compute(Redundant_matrix.leftCols(redundant_size + 1));
58
59 // Check if the new column is linearly independent to other ones
60 if(Problem.Solver.qr.rank() == redundant_size + 1){
61 Permutation_trip.push_back(Trip(var_iter, Problem.Variables_num - Problem.Constraints_eq_num + redundant_size, 1));
62 redundant_size += 1;
63 }
64 else{
65 // If not indepedent, remove the current column
66 Subspan_matrix.coeffRef(var_iter, redundant_size) = 0;
67 Permutation_trip.push_back(Trip(var_iter, remaining_size, 1));
68 remaining_size += 1;
69 }
70 }
71 else{
72 Permutation_trip.push_back(Trip(var_iter, remaining_size, 1));
73 remaining_size += 1;
74 }
75 }
76 Problem.Constraint.permutation_matrix.setFromTriplets(Permutation_trip.begin(), Permutation_trip.end());
77
78 // Update the constraints, objective function, and initial feasible solution with the permutation
79 Problem.Constraint.eq_orig_matrix = Problem.Constraint.eq_orig_matrix * Problem.Constraint.permutation_matrix;
80 Problem.Boundary.ie_orig_matrix = Problem.Constraint.permutation_matrix.transpose() * Problem.Boundary.ie_orig_matrix;
81 Problem.Objective.orig_vector = Problem.Constraint.permutation_matrix.transpose() * Problem.Objective.orig_vector;
82 if(stepwise_obj){
83 Problem.Objective.varying_vector = Problem.Constraint.permutation_matrix.transpose() * Problem.Objective.varying_vector;
84 }
85}

102void LP_constraint_ie_reduced_generation(LP_object &Problem){
103 // Set default value of reduced constraint matrix
104 Eigen::MatrixXd reduced_matrix = Eigen::MatrixXd(Problem.Constraint.ie_orig_matrix.leftCols(Problem.Variables_num - Problem.Constraints_eq_num));
105
106 // Check if there are eqaulity constraints in the original problem
107 if(Problem.Constraints_eq_num != 0){
108 // If equality constraints exist, replacing the redundant variables with them
109 // A_e * x_e + A_r * x_r = b
110 // A_e * x_e - b = -A_r * x_r
111 // l_r <= -A_r^(-1) * (A_e * x_e - b) = x_r <= u_r
112 // l_r - A_r^(-1) * b <= -A_r^(-1) * A_e * x_e <= u_r - A_r^(-1) * b
113 Eigen::VectorXd col_orig;
114 // Do not use openmp parallel thread to run the loop below; may yield wrong solution
115 for(int var_id = 0; var_id < Problem.Variables_num - Problem.Constraints_eq_num; ++ var_id){
116 col_orig = Problem.Constraint.eq_orig_matrix.col(var_id);
117 reduced_matrix.middleRows(Problem.Variables_num - Problem.Constraints_eq_num, Problem.Constraints_eq_num).col(var_id) = -Problem.Solver.lu.solve(col_orig);
118 }
119
120 // Check if there are ineqaulity constraints in the original problem
121 if(Problem.Constraints_ie_num != 0){
122 // If inequality constraints exist in the original, renew the coefficients
123 // l_i <= M_e * x_e + M_r * x_r <= u_i
124 // l_i <= M_e * x_e - M_r * A_r^(-1) * (A_e * x_e - b) <= u_i
125 // l_i - M_r * A_r^(-1) * b <= (M_e - M_r * A_r^(-1) * A_e) * x_e <= u_i - M_r * A_r^(-1) * b
126 }
127 }
128
129 double tol = pow(10, -12);
130 Problem.Constraint.ie_reduced_matrix = reduced_matrix.sparseView(tol, 1);
131}

133void LP_boundary_ie_reduced_generation(LP_object &Problem){
134 // Set default value of reduced boundary vector
135 Problem.Boundary.ie_reduced_matrix = Problem.Boundary.ie_orig_matrix;
136
137 // Check if there are eqaulity constraints in the original problem
138 if(Problem.Constraints_eq_num != 0){
139 // If equality constraints exist, replacing the redundant variables with them
140 // A_e * x_e + A_r * x_r = b
141 // A_e * x_e - b = -A_r * x_r
142 // l_r <= -A_r^(-1) * (A_e * x_e - b) = x_r <= u_r
143 // l_r - A_r^(-1) * b <= -A_r^(-1) * A_e * x_e <= u_r - A_r^(-1) * b
144 Problem.Boundary.ie_reduced_matrix.middleRows(Problem.Variables_num - Problem.Constraints_eq_num, Problem.Constraints_eq_num).col(0) -= Problem.Solver.lu.solve(Problem.Boundary.eq_vector);
145 Problem.Boundary.ie_reduced_matrix.middleRows(Problem.Variables_num - Problem.Constraints_eq_num, Problem.Constraints_eq_num).col(1) -= Problem.Solver.lu.solve(Problem.Boundary.eq_vector);
146
147 // Check if there are ineqaulity constraints in the original problem
148 if(Problem.Constraints_ie_num != 0){
149 // If inequality constraints exist in the original, renew the coefficients
150 // l_i <= M_e * x_e + M_r * x_r <= u_i
151 // l_i <= M_e * x_e - M_r * A_r^(-1) * (A_e * x_e - b) <= u_i
152 // l_i - M_r * A_r^(-1) * b <= (M_e - M_r * A_r^(-1) * A_e) * x_e <= u_i - M_r * A_r^(-1) * b
153 }

157void LP_objective_reduced_generation(LP_object &Problem){
158 // Set default value of reduced objective vector
159 Problem.Objective.reduced_vector = Problem.Objective.orig_vector.head(Problem.Variables_num - Problem.Constraints_eq_num);
160
161 // Check if there are eqaulity constraints in the original problem
162 if(Problem.Constraints_eq_num != 0){
163 // Z = c_e^T * x_e + c_r^T * x_r
164 // = c_e^T * x_e - c_r^T * A_r^(-1) * (A_e * x_e - b)
165 // = (c_e^T - c_r^T * A_r^(-1) * A_e) * x_e + c_r^T * A_r^(-1) * b
166 Problem.Objective.reduced_vector = Problem.Objective.orig_vector.head(Problem.Variables_num - Problem.Constraints_eq_num);
167 Eigen::VectorXd coeff_redundant = Problem.Objective.orig_vector.tail(Problem.Constraints_eq_num);
168 Problem.Objective.reduced_vector -= (Problem.Constraint.eq_orig_matrix.leftCols(Problem.Variables_num - Problem.Constraints_eq_num)).transpose() * Problem.Solver.lu.transpose().solve(coeff_redundant);
169 }

479 // Reconstruction of the objective value
480 // Z = c_e^T * x_e + c_r^T * x_r
481 // = c_e^T * x_e - c_r^T * A_r^(-1) * (A_e * x_e - b)
482 // = (c_e^T - c_r^T * A_r^(-1) * A_e) * x_e + c_r^T * A_r^(-1) * b
483 Problem.Objective.orig_value = Problem.Objective.reduced_value + Problem.Objective.orig_vector.tail(Problem.Constraints_eq_num).transpose() * Problem.Solver.lu.solve(Problem.Boundary.eq_vector);
484 }
485 else{
486 Problem.Solution.orig_vector = Problem.Solution.reduced_vector;
487 Problem.Objective.orig_value = Problem.Objective.reduced_value;
488 }
489
490 // Permute the original solution to the correct order
491 Problem.Solution.orig_vector = Problem.Constraint.permutation_matrix * Problem.Solution.orig_vector;
492}
