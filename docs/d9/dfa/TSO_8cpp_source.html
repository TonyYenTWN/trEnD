22 int to_ID = Power_network_inform.edges.to(edge_iter);
23 int voltage = Power_network_inform.edges.voltage_base(edge_iter);

37 power_limit.reserve(TSO_Market.network.num_vertice * TSO_Market.network.num_vertice);
38 for(int row_iter = 0; row_iter < TSO_Market.network.num_vertice - 1; ++ row_iter){
39 for(int col_iter = row_iter + 1; col_iter < TSO_Market.network.num_vertice; ++ col_iter){
40 if(abs(admittance(row_iter , col_iter)) > tol){
41 TSO_Market.network.incidence.push_back(Eigen::Vector2i(row_iter, col_iter));
42 TSO_Market.network.admittance.push_back(admittance(row_iter , col_iter));
43 power_limit.push_back(capacity(row_iter , col_iter));

51 TSO_Market.network.voltage_constraint.col(0) *= -Power_network_inform.tech_parameters.theta_limit;
52 TSO_Market.network.voltage_constraint.col(1) *= Power_network_inform.tech_parameters.theta_limit;
53 TSO_Market.network.power_constraint = Eigen::MatrixXd (TSO_Market.network.num_edges, 2);

55 TSO_Market.network.power_constraint.col(1) /= Power_network_inform.tech_parameters.s_base;
56 TSO_Market.network.power_constraint.col(0) = -TSO_Market.network.power_constraint.col(1);

72void power_market::TSO_Market_Results_Get(int tick, market_inform &Market, alglib::minlpstate &Problem){
73 alglib::real_1d_array sol;
74 alglib::minlpreport rep;
75 alglib::minlpresults(Problem, sol, rep);
76 Eigen::VectorXd sol_vec = Eigen::Map <Eigen::VectorXd> (sol.getcontent(), sol.length());
77
78 for(int node_iter = 0; node_iter < Market.network.num_vertice; ++ node_iter){
79 // Store power source / sink
80 int row_start = 2 * Market.network.num_vertice + node_iter * (Market.price_intervals + 2);
81 Market.confirmed_supply(tick, node_iter) = (sol_vec.segment(row_start, Market.price_intervals + 2).array().max(0)).sum();
82 Market.confirmed_demand(tick, node_iter) = -(sol_vec.segment(row_start, Market.price_intervals + 2).array().min(0)).sum();
83
84 // Store nodal prices
85 Market.confirmed_price(tick, node_iter) = Market.bidded_price(0) + rep.lagbc[row_start];
86 Market.confirmed_price(tick, node_iter) = std::min(Market.confirmed_price(tick, node_iter), Market.price_range_inflex(1));
87 Market.confirmed_price(tick, node_iter) = std::max(Market.confirmed_price(tick, node_iter), Market.price_range_inflex(0));
88
89 // Store ratio at nodes
90 for(int price_iter = 0; price_iter < Market.price_intervals + 2; ++ price_iter){
91 if(Market.bidded_price(price_iter) >= Market.confirmed_price(tick, node_iter) || price_iter == Market.price_intervals + 1){
92 Market.confirmed_price_ratio(tick, node_iter) = sol_vec(row_start + price_iter);
93 if(sol_vec(row_start + price_iter) >= 0.){
94 Market.confirmed_price_ratio(tick, node_iter) /= Market.submitted_supply(price_iter, node_iter);
95 }
96 else{
97 Market.confirmed_price_ratio(tick, node_iter) /= Market.submitted_demand(price_iter, node_iter);
98 }
99 break;
100 }
101 }
102
103 // Store voltage and power flow
104 Market.network.confirmed_voltage.row(tick) = sol_vec.head(Market.network.num_vertice);
105 Market.network.confirmed_power.row(tick) = sol_vec.tail(Market.network.num_edges);
106 }
107
108 std::cout << sol_vec.segment(Market.network.num_vertice, Market.network.num_vertice).minCoeff() << " " << sol_vec.segment(Market.network.num_vertice, Market.network.num_vertice).maxCoeff() << " " << .5 * sol_vec.segment(Market.network.num_vertice, Market.network.num_vertice).array().abs().sum() << "\n";
109 std::cout << sol_vec.head(Market.network.num_vertice).minCoeff() << " " << sol_vec.head(Market.network.num_vertice).maxCoeff() << "\n";
110 std::cout << sol_vec.tail(Market.network.num_edges).minCoeff() << " " << sol_vec.tail(Market.network.num_edges).maxCoeff() << "\n\n";
