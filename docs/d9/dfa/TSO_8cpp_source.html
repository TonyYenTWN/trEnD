19void power_market::TSO_Market_Set(market_inform &TSO_Market, power_network::network_inform &Power_network_inform, int Time){
20// double pi = boost::math::constants::pi<double>();
21
22 // Input parameters of TSO market
23 TSO_Market.num_zone = Power_network_inform.nodes.bidding_zone.size();
24 TSO_Market.time_intervals = Time;
25 TSO_Market.set_bidded_price();
26
27 // Set node admittance matrix and line capacity matrix
28 TSO_Market.network.num_vertice = TSO_Market.num_zone;
29 Eigen::MatrixXd admittance = Eigen::MatrixXd::Zero(TSO_Market.network.num_vertice, TSO_Market.network.num_vertice);
30 Eigen::MatrixXd capacity = Eigen::MatrixXd::Zero(TSO_Market.network.num_vertice, TSO_Market.network.num_vertice);
31 for(int edge_iter = 0; edge_iter < Power_network_inform.edges.distance.size(); ++ edge_iter){
32 int from_ID = Power_network_inform.edges.from(edge_iter );
33 int to_ID = Power_network_inform.edges.to(edge_iter);
34 int voltage = Power_network_inform.edges.voltage_base(edge_iter);
35 double y_ij = 1. / Power_network_inform.edges.distance(edge_iter);
36 y_ij /= Power_network_inform.tech_parameters.z_distr_series.imag();
37 y_ij *= Power_network_inform.tech_parameters.impedenace_base_levels[voltage];
38 admittance(from_ID, to_ID) += y_ij ;
39 admittance(to_ID, from_ID) += y_ij ;
40 capacity(from_ID, to_ID) += Power_network_inform.tech_parameters.power_limit[voltage];
41 }
42
43 // Set compact incidence matrix and edge admittance matrix
44 double tol = 1E-6;
45 TSO_Market.network.incidence.reserve(TSO_Market.network.num_vertice * TSO_Market.network.num_vertice);
46 TSO_Market.network.admittance.reserve(TSO_Market.network.num_vertice * TSO_Market.network.num_vertice);
47 std::vector <double> power_limit;
48 power_limit.reserve(TSO_Market.network.num_vertice * TSO_Market.network.num_vertice);
49 for(int row_iter = 0; row_iter < TSO_Market.network.num_vertice - 1; ++ row_iter){
50 for(int col_iter = row_iter + 1; col_iter < TSO_Market.network.num_vertice; ++ col_iter){
51 if(abs(admittance(row_iter , col_iter)) > tol){
52 TSO_Market.network.incidence.push_back(Eigen::Vector2i(row_iter, col_iter));
53 TSO_Market.network.admittance.push_back(admittance(row_iter , col_iter));
54 power_limit.push_back(capacity(row_iter , col_iter));
55 }
56 }
57 }
58 TSO_Market.network.num_edges = TSO_Market.network.incidence.size();
59
60 // Set voltage and power constraints at each edge
61 TSO_Market.network.voltage_constraint = Eigen::MatrixXd::Ones(TSO_Market.network.num_vertice, 2);
62 TSO_Market.network.voltage_constraint.col(0) *= -Power_network_inform.tech_parameters.theta_limit;
63 TSO_Market.network.voltage_constraint.col(1) *= Power_network_inform.tech_parameters.theta_limit;
64 TSO_Market.network.power_constraint = Eigen::MatrixXd (TSO_Market.network.num_edges, 2);
65 TSO_Market.network.power_constraint.col(1) = Eigen::Map <Eigen::VectorXd> (power_limit.data(), power_limit.size());
66 TSO_Market.network.power_constraint.col(1) /= Power_network_inform.tech_parameters.s_base;
67 TSO_Market.network.power_constraint.col(0) = -TSO_Market.network.power_constraint.col(1);
68
69 // Initialization of process variables
70 power_market::Market_Initialization(TSO_Market);
71
72 // Initialization of output variables
73 TSO_Market.confirmed_supply = Eigen::MatrixXd::Zero(Time, TSO_Market.num_zone);
74 TSO_Market.confirmed_demand = Eigen::MatrixXd::Zero(Time, TSO_Market.num_zone);
75 TSO_Market.confirmed_price = Eigen::MatrixXd(Time, TSO_Market.num_zone);
76 TSO_Market.network.confirmed_voltage = Eigen::MatrixXd(Time, TSO_Market.network.num_vertice);
77 TSO_Market.network.confirmed_power = Eigen::MatrixXd(Time, TSO_Market.network.num_edges);
78}

80void power_market::TSO_Market_Results_Get(int tick, market_inform &Market, alglib::minlpstate &Problem){
81 alglib::real_1d_array sol;
82 alglib::minlpreport rep;
83 alglib::minlpresults(Problem, sol, rep);

85
86 for(int node_iter = 0; node_iter < Market.network.num_vertice; ++ node_iter){
87 // Store power source / sink
88 int row_start = 2 * Market.network.num_vertice + node_iter * (Market.price_intervals + 2);
89 Market.confirmed_supply(tick, node_iter) = (sol_vec.segment(row_start, Market.price_intervals + 2).array().max(0)).sum();
90 Market.confirmed_demand(tick, node_iter) = -(sol_vec.segment(row_start, Market.price_intervals + 2).array().min(0)).sum();
91
92 // Store nodal prices
93 Market.confirmed_price(tick, node_iter) = Market.bidded_price(0) + rep.lagbc[row_start];
94 Market.confirmed_price(tick, node_iter) = std::min(Market.confirmed_price(tick, node_iter), Market.price_range_inflex(1));
95 Market.confirmed_price(tick, node_iter) = std::max(Market.confirmed_price(tick, node_iter), Market.price_range_inflex(0));
96
97 // Store voltage and power flow
98 Market.network.confirmed_voltage.row(tick) = sol_vec.head(Market.network.num_vertice);
99 Market.network.confirmed_power.row(tick) = sol_vec.tail(Market.network.num_edges);
100 }
101
102 std::cout << sol_vec.segment(Market.network.num_vertice, Market.network.num_vertice).minCoeff() << " " << sol_vec.segment(Market.network.num_vertice, Market.network.num_vertice).maxCoeff() << " " << .5 * sol_vec.segment(Market.network.num_vertice, Market.network.num_vertice).array().abs().sum() << "\n";
103 std::cout << sol_vec.head(Market.network.num_vertice).minCoeff() << " " << sol_vec.head(Market.network.num_vertice).maxCoeff() << "\n";
104 std::cout << sol_vec.tail(Market.network.num_edges).minCoeff() << " " << sol_vec.tail(Market.network.num_edges).maxCoeff() << "\n\n";
105}
