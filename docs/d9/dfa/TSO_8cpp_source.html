32 // Set compact incidence matrix and edge admittance matrix

36 std::vector <double> power_limit;
37 power_limit.reserve(TSO_Market.network.num_vertice * TSO_Market.network.num_vertice);

42 TSO_Market.network.admittance.push_back(admittance(row_iter , col_iter));

52 TSO_Market.network.voltage_constraint.col(1) *= Power_network_inform.tech_parameters.theta_limit;

83
84 for(int node_iter = 0; node_iter < Market.network.num_vertice; ++ node_iter){
85 // Store power source / sink
86 int row_start = 2 * Market.network.num_vertice + node_iter * (Market.price_intervals + 2);
87 Market.confirmed_supply(tick, node_iter) = (sol_vec.segment(row_start, Market.price_intervals + 2).array().max(0)).sum();
88 Market.confirmed_demand(tick, node_iter) = -(sol_vec.segment(row_start, Market.price_intervals + 2).array().min(0)).sum();
89
90 // Store nodal prices
91 Market.confirmed_price(tick, node_iter) = Market.bidded_price(0) + rep.lagbc[row_start];
92 Market.confirmed_price(tick, node_iter) = std::min(Market.confirmed_price(tick, node_iter), Market.price_range_inflex(1));
93 Market.confirmed_price(tick, node_iter) = std::max(Market.confirmed_price(tick, node_iter), Market.price_range_inflex(0));
94
95 // Store ratio at nodes
96 for(int price_iter = 0; price_iter < Market.price_intervals + 2; ++ price_iter){
97 if(Market.bidded_price(price_iter) >= Market.confirmed_price(tick, node_iter) || price_iter == Market.price_intervals + 1){
98 Market.confirmed_price_ratio(tick, node_iter) = sol_vec(row_start + price_iter);
99 if(sol_vec(row_start + price_iter) >= 0.){
100 Market.confirmed_price_ratio(tick, node_iter) /= Market.submitted_supply(price_iter, node_iter);
101 }
102 else{
103 Market.confirmed_price_ratio(tick, node_iter) /= Market.submitted_demand(price_iter, node_iter);
104 }
105 break;
106 }
107 }
108
109 // Store voltage and power flow
110 Market.network.confirmed_voltage.row(tick) = sol_vec.head(Market.network.num_vertice);
111 Market.network.confirmed_power.row(tick) = sol_vec.tail(Market.network.num_edges);
112 }
