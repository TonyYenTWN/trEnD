259 // Set the maps for levels of voltage and impedance base
260 void set_level_maps(nodes &nodes){
261 int voltage_min = nodes.voltage_base.minCoeff();
262 int voltage_max = voltage_min;
263 int level_count = 0;
264 this->voltage_base_levels.insert(std::make_pair(voltage_max, level_count));
265 this->impedenace_base_levels.insert(std::make_pair(voltage_max, voltage_max * voltage_max / this->s_base));
266 this->power_limit.insert(std::make_pair(voltage_max, (double) voltage_max));
267
268 std::vector <int> voltage_base_sorted(nodes.voltage_base.data(), nodes.voltage_base.data() + nodes.voltage_base.size());
269 std::sort(voltage_base_sorted.begin(), voltage_base_sorted.end());
270 for(int node_iter = 0; node_iter < nodes.voltage_base.size(); ++ node_iter){
271 if(voltage_base_sorted[node_iter] > voltage_max){
272 voltage_max = voltage_base_sorted[node_iter];
273 level_count += 1;
274 this->voltage_base_levels.insert(std::make_pair(voltage_max, level_count));
275 this->impedenace_base_levels.insert(std::make_pair(voltage_max, (double) voltage_max * voltage_max / this->s_base / 3.));
276 this->power_limit.insert(std::make_pair(voltage_max, (double) voltage_max));
277 //std::cout << voltage_base_levels[voltage_max] << "\t" << voltage_max << "\t" << impedenace_base_levels[level_count] << "\n";
