256 // Set the maps for levels of voltage and impedance base
257 void set_level_maps(nodes &nodes){
258 int voltage_min = nodes.voltage_base.minCoeff();
259 int voltage_max = voltage_min;
260 int level_count = 0;
261 this->voltage_base_levels.insert(std::make_pair(voltage_max, level_count));
262 this->impedenace_base_levels.insert(std::make_pair(voltage_max, (double) voltage_max * voltage_max / this->s_base / 3.));
263 this->power_limit.insert(std::make_pair(voltage_max, (double) voltage_max));
264
265 std::vector <int> voltage_base_sorted(nodes.voltage_base.data(), nodes.voltage_base.data() + nodes.voltage_base.size());
266 std::sort(voltage_base_sorted.begin(), voltage_base_sorted.end());
267 for(int node_iter = 0; node_iter < nodes.voltage_base.size(); ++ node_iter){
268 if(voltage_base_sorted[node_iter] > voltage_max){
269 voltage_max = voltage_base_sorted[node_iter];
270 level_count += 1;
271 this->voltage_base_levels.insert(std::make_pair(voltage_max, level_count));
272 this->impedenace_base_levels.insert(std::make_pair(voltage_max, (double) voltage_max * voltage_max / this->s_base / 3.));
273 this->power_limit.insert(std::make_pair(voltage_max, (double) voltage_max));
274 //std::cout << voltage_base_levels[voltage_max] << "\t" << voltage_max << "\t" << impedenace_base_levels[level_count] << "\n";
275 }
276 }
